```typescript

  type: ArtifactType,
  params?: Record<string, string>   // NEW — optional search/filter params

```

Inside the `useEffect` that fetches, change the fetch URL:

```typescript
// Build URL with optional params
const url = params && Object.keys(params).length > 0
  ? `${basePath}?${new URLSearchParams(params).toString()}`
  : basePath;

fetch(url, { cache: "no-store" })
```

Add `params` to the `useEffect` dependency array:

```typescript
useEffect(() => {
  // ... existing effect body ...
}, [basePath, refreshKey, params]);   // add params here
```

**IMPORTANT:** `params` is an object — adding it directly to the dep array causes infinite loops because object identity changes on every render. The caller must memoize it with `useMemo`. Document this in the hook's JSDoc:

```typescript
/**
 * @param params - Optional search/filter params. MUST be memoized with useMemo
 *   at the call site to prevent infinite fetch loops.
 *   Example: const params = useMemo(() => ({ q, category }), [q, category])
 */
```

**Step 1: Create `use-user-skills.ts` hook**

```typescript
"use client";
/**
 * useUserSkills — fetches active skills visible to the current user.
 *
 * Calls GET /api/skills with optional q, category, sort params.
 * The Next.js proxy at /api/skills proxies to backend /api/skills
 * with the user's JWT injected server-side.
 */
import { useState, useEffect, useCallback } from "react";

export interface UserSkill {
  id: string;
  name: string;
  displayName: string | null;
  description: string | null;
  version: string;
  skillType: string;
  slashCommand: string | null;
  category: string | null;
  tags: string[] | null;
  license: string | null;
}

interface UseUserSkillsOptions {
  q?: string;
  category?: string;
  sort?: "name" | "created_at" | "updated_at";
}

export function useUserSkills(options: UseUserSkillsOptions = {}) {
  const [skills, setSkills] = useState<UserSkill[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetch_ = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const params = new URLSearchParams();
      if (options.q?.trim()) params.set("q", options.q.trim());
      if (options.category) params.set("category", options.category);
      if (options.sort) params.set("sort", options.sort);
      const url = `/api/skills${params.size ? `?${params}` : ""}`;
      const res = await fetch(url, { cache: "no-store" });
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      const data = (await res.json()) as Record<string, unknown>[];
      // Backend returns snake_case — convert to camelCase for the fields we need
      setSkills(
        data.map((s) => ({
          id: s.id as string,
          name: s.name as string,
          displayName: (s.display_name ?? null) as string | null,
          description: (s.description ?? null) as string | null,
          version: (s.version ?? "1.0.0") as string,
          skillType: s.skill_type as string,
          slashCommand: (s.slash_command ?? null) as string | null,
          category: (s.category ?? null) as string | null,
          tags: (s.tags ?? null) as string[] | null,
          license: (s.license ?? null) as string | null,
        }))
      );
    } catch (e) {
      setError(e instanceof Error ? e.message : "Failed to load skills");
    } finally {
      setLoading(false);
    }
  }, [options.q, options.category, options.sort]);

  useEffect(() => {
    void fetch_();
  }, [fetch_]);

  return { skills, loading, error, refetch: fetch_ };
}
```

/**
 * Next.js proxy: GET /api/skills → backend GET /api/skills
 * Injects JWT from the server-side session.
 */
import { auth } from "@/auth";
import { NextResponse } from "next/server";

const BACKEND_URL = process.env.BACKEND_URL ?? process.env.NEXT_PUBLIC_API_URL ?? "http://backend:8000";

export async function GET(request: Request): Promise<NextResponse> {
  const session = await auth();
  const token = (session as Record<string, unknown>)?.accessToken as string | undefined;

  const { searchParams } = new URL(request.url);
  const backendUrl = `${BACKEND_URL}/api/skills?${searchParams.toString()}`;

  const res = await fetch(backendUrl, {
    headers: token ? { Authorization: `Bearer ${token}` } : {},
    cache: "no-store",
  });

  const data: unknown = await res.json();
  return NextResponse.json(data, { status: res.status });
}

**Step 3: Replace `/skills/page.tsx` stub with the catalog**

```typescript
"use client";
/**
 * Skills catalog — active skills visible to the current user.
 * Search by name/description, filter by category, sort by name or date.
 */
import { useState, useMemo } from "react";
import { useUserSkills } from "@/hooks/use-user-skills";

function useDebounce(value: string, delay: number): string {
  const [debounced, setDebounced] = useState(value);
  useMemo(() => {
    const timer = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(timer);
  }, [value, delay]);
  return debounced;
}

export default function SkillsPage() {
  const [query, setQuery] = useState("");
  const [selectedCategory, setSelectedCategory] = useState<string | null>(null);
  const [sort, setSort] = useState<"name" | "updated_at">("updated_at");

  const debouncedQuery = useDebounce(query, 300);

  const { skills, loading, error } = useUserSkills({
    q: debouncedQuery || undefined,
    category: selectedCategory ?? undefined,
    sort,
  });

  // Derive unique categories from results
  const categories = useMemo(() => {
    const cats = new Set(skills.map((s) => s.category).filter(Boolean) as string[]);
    return Array.from(cats).sort();
  }, [skills]);

  return (
    <div className="p-6 max-w-5xl mx-auto space-y-6">
      {/* Header */}
      <div>
        <h1 className="text-2xl font-semibold text-gray-900">Skills</h1>
        <p className="text-sm text-gray-500 mt-1">
          Browse and run skills available to you.
        </p>
      </div>

      {/* Search + sort bar */}
      <div className="flex flex-col sm:flex-row gap-3">
        <div className="relative flex-1">
          <input
            type="text"
            placeholder="Search skills..."
            value={query}
            onChange={(e) => setQuery(e.target.value)}
            className="w-full pl-9 pr-4 py-2 text-sm border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
          />
          <svg
            className="absolute left-3 top-2.5 h-4 w-4 text-gray-400"
            fill="none"
            viewBox="0 0 24 24"
            stroke="currentColor"
          >
            <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2}
              d="M21 21l-6-6m2-5a7 7 0 11-14 0 7 7 0 0114 0z" />
          </svg>
        </div>
        <select
          value={sort}
          onChange={(e) => setSort(e.target.value as "name" | "updated_at")}
          className="text-sm border border-gray-300 rounded-lg px-3 py-2 focus:outline-none focus:ring-2 focus:ring-blue-500 bg-white"
        >
          <option value="updated_at">Newest first</option>
          <option value="name">Name A–Z</option>
        </select>
      </div>

      {/* Category chips */}
      {categories.length > 0 && (
        <div className="flex flex-wrap gap-2">
          <button
            onClick={() => setSelectedCategory(null)}
            className={`px-3 py-1 text-xs rounded-full border transition-colors ${
              selectedCategory === null
                ? "bg-blue-600 text-white border-blue-600"
                : "border-gray-300 text-gray-600 hover:border-blue-400"
            }`}
          >
            All
          </button>
          {categories.map((cat) => (
            <button
              key={cat}
              onClick={() => setSelectedCategory(selectedCategory === cat ? null : cat)}
              className={`px-3 py-1 text-xs rounded-full border transition-colors ${
                selectedCategory === cat
                  ? "bg-blue-600 text-white border-blue-600"
                  : "border-gray-300 text-gray-600 hover:border-blue-400"
              }`}
            >
              {cat}
            </button>
          ))}
        </div>
      )}

      {/* States */}
      {loading && (
        <div className="text-center py-16 text-gray-500 text-sm">Loading skills...</div>
      )}
      {error && (
        <div className="rounded-md bg-red-50 border border-red-200 px-4 py-3 text-sm text-red-700">
          {error}
        </div>
      )}
      {!loading && !error && skills.length === 0 && (
        <div className="text-center py-16 text-gray-500 text-sm">
          {debouncedQuery || selectedCategory
            ? "No skills match your search."
            : "No skills available yet."}
        </div>
      )}

      {/* Skills grid */}
      {!loading && !error && skills.length > 0 && (
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
          {skills.map((skill) => (
            <div
              key={skill.id}
              className="bg-white border border-gray-200 rounded-lg p-4 hover:shadow-md transition-shadow flex flex-col"
            >
              <div className="flex items-start justify-between gap-2 mb-2">
                <h3 className="font-semibold text-gray-900 text-sm leading-tight">
                  {skill.displayName ?? skill.name}
                </h3>
                <span className="shrink-0 text-xs bg-gray-100 text-gray-600 px-2 py-0.5 rounded-full font-mono">
                  v{skill.version}
                </span>
              </div>
              {skill.description && (
                <p className="text-xs text-gray-600 leading-relaxed mb-3 flex-1">
                  {skill.description}
                </p>
              )}
              <div className="flex flex-wrap gap-1 mt-auto">
                {skill.category && (
                  <span className="text-xs px-2 py-0.5 bg-blue-50 text-blue-700 rounded-full">
                    {skill.category}
                  </span>
                )}
                {skill.slashCommand && (
                  <span className="text-xs px-2 py-0.5 bg-gray-100 text-gray-600 rounded-full font-mono">
                    {skill.slashCommand}
                  </span>
                )}
                {skill.tags?.map((tag) => (
                  <span key={tag} className="text-xs px-2 py-0.5 bg-gray-50 text-gray-500 rounded-full">
                    {tag}
                  </span>
                ))}
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```

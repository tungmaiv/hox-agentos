**Goal:** Move external registry browse/import from `user_router` (chat permission) to `admin_router` (registry:manage), add pagination to the browse endpoint, and expand the import dialog to show full skill metadata before confirming the import.

**Architecture:** Three changes to the same feature area:
1. **Security fix** — move routes, update frontend URLs (`/api/skill-repos/...` → `/api/admin/skill-repos/...`)
2. **Pagination** — `browse_skills()` service slices the cached index list; new `BrowseResponse` schema wraps items with total/page/has_more; frontend appends pages with "Load more"
3. **Detail preview** — add explicit metadata fields to `SkillBrowseItem`, expand the existing `"confirm"` dialog phase into a `"detail"` phase showing full metadata before the confirm button

**Tech Stack:** FastAPI, Pydantic, SQLAlchemy async for the backend; React, TypeScript strict, Tailwind for the frontend.

```python
async def browse_skills(
    query: str | None,
    session: AsyncSession,
    page: int = 1,
    limit: int = 20,
) -> "BrowseResponse":
    """Aggregate skills from all active repositories with optional search and pagination.

    Reads cached_index from each active SkillRepository. No remote requests.

    Args:
        query: Optional search string — case-insensitive substring match on name/description.
        session: Async DB session.
        page: Page number (1-based).
        limit: Items per page (default 20).

    Returns:
        BrowseResponse with paginated items, total count, and has_more flag.
    """
    from skill_repos.schemas import BrowseResponse  # avoid circular at module level

    result = await session.execute(
        select(SkillRepository).where(SkillRepository.is_active == True)  # noqa: E712
    )
    repos = result.scalars().all()

    all_items: list[SkillBrowseItem] = []
    for repo in repos:
        if not repo.cached_index:
            continue
        skills: list[dict[str, Any]] = repo.cached_index.get("skills", [])
        for skill in skills:
            name: str = skill.get("name", "")
            description: str | None = skill.get("description") or None

            if query:
                q = query.lower()
                if q not in name.lower() and q not in (description or "").lower():
                    continue

            meta: dict[str, Any] = skill.get("metadata") or {}
            all_items.append(
                SkillBrowseItem(
                    name=name,
                    description=description,
                    version=skill.get("version"),
                    repository_name=repo.name,
                    repository_id=str(repo.id),
                    metadata=meta,
                    # Explicit fields extracted for detail preview
                    license=meta.get("license") or skill.get("license"),
                    tags=skill.get("tags") or meta.get("tags"),
                    category=skill.get("category") or meta.get("category"),
                    compatibility=skill.get("compatibility") or meta.get("compatibility"),
                    source_url=skill.get("source_url") or meta.get("source_url"),
                    author=meta.get("author"),
                )
            )

    total = len(all_items)
    start = (page - 1) * limit
    paginated = all_items[start : start + limit]

    return BrowseResponse(
        items=paginated,
        total=total,
        page=page,
        limit=limit,
        has_more=(start + limit) < total,
    )
```

async def browse_skills_route(
    q: str | None = Query(None, description="Search by name or description"),
    page: int = Query(1, ge=1, description="Page number (1-based)"),
    limit: int = Query(20, ge=1, le=100, description="Items per page"),
    user: UserContext = Depends(require_registry_manager),
    session: AsyncSession = Depends(get_db),
) -> BrowseResponse:
    """Browse skills from all active registered repositories (admin only).

    Reads from cached index — no remote requests. Supports search and pagination.
    """
    result = await browse_skills(q, session, page=page, limit=limit)
    logger.info(
        "admin_skill_repos_browse",
        user_id=str(user["user_id"]),
        query=q,
        page=page,
        result_count=len(result.items),
        total=result.total,
    )
    return result


@admin_router.post("/import", response_model=ImportResponse, status_code=201)
async def import_skill_route(
    body: ImportRequest,
    user: UserContext = Depends(require_registry_manager),
    session: AsyncSession = Depends(get_db),
) -> ImportResponse:
    """Import a skill from a registered repository (admin only).

    The skill enters pending_review status and must be approved before becoming active.
    A security scan runs automatically on import.
    """
    try:
        repo_uuid = UUID(body.repository_id)
    except ValueError as exc:
        raise HTTPException(
            status_code=422, detail="Invalid repository_id — must be a UUID"

    result = await import_from_repo(
        repo_id=repo_uuid,
        skill_name=body.skill_name,
        user_id=user["user_id"],
        session=session,
    )
    logger.info(
        "admin_skill_repo_import",
        repo_id=body.repository_id,
        skill_name=body.skill_name,
        skill_id=result.skill_id,
        user_id=str(user["user_id"]),
    )
    return result

const fetchSkills = useCallback(async (q: string, pageNum: number = 1, append: boolean = false) => {
  setLoading(true);
  setError(null);
  try {
    const params = new URLSearchParams({ page: String(pageNum), limit: "20" });
    if (q.trim()) params.set("q", q.trim());
    const response = await fetch(`/api/admin/skill-repos/browse?${params}`);
    if (!response.ok) {
      const data = (await response.json().catch(() => ({}))) as { detail?: string };
      setError(data.detail ?? `Error ${response.status}`);
      return;
    }
    const data = (await response.json()) as BrowseResponse;
    setSkills((prev) => append ? [...prev, ...data.items] : data.items);
    setHasMore(data.has_more);
    setTotal(data.total);
    setPage(pageNum);
  } catch {
    setError("Failed to connect to backend");
  } finally {
    setLoading(false);
  }
}, []);

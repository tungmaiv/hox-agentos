```python
"""
Sandbox execution policies.

Defines per-language resource limits, base images, and timeouts.
"""
from dataclasses import dataclass


@dataclass
class SandboxPolicy:
    """Resource constraints for a sandbox container."""

    image: str
    nano_cpus: int = 500_000_000   # 0.5 CPU
    mem_limit: str = "128m"
    memswap_limit: str = "128m"    # mem + swap; equal = no swap
    network_mode: str = "none"     # no internet access
    timeout_seconds: int = 30
    read_only: bool = True         # no writes to container FS


_POLICIES: dict[str, SandboxPolicy] = {
    "python": SandboxPolicy(image="blitz-sandbox-python:latest"),
    "bash": SandboxPolicy(image="bash:5", timeout_seconds=15),
}


def get_policy(lang: str) -> SandboxPolicy:
    """Return the SandboxPolicy for the given language.

    Raises ValueError for unsupported languages.
    """
    policy = _POLICIES.get(lang)
    if policy is None:
        supported = ", ".join(sorted(_POLICIES.keys()))
        raise ValueError(f"Unsupported language: {lang!r}. Supported: {supported}")
    return policy
```

### Step 10: Create `backend/sandbox/executor.py`

```python
"""
Sandbox executor — orchestrates container lifecycle for code execution.

Flow: validate policy → create container → exec code → capture output → destroy
"""
import asyncio
import time
from dataclasses import dataclass
from typing import Any

import docker
import structlog

from sandbox.policies import SandboxPolicy, get_policy

logger = structlog.get_logger(__name__)

_CONTAINER_NAME_PREFIX = "blitz-sandbox-"


@dataclass
class SandboxResult:
    """Output from a sandbox execution run."""

    stdout: str
    stderr: str
    exit_code: int
    duration_ms: int
    timed_out: bool


class SandboxExecutor:
    """Executes code in an ephemeral Docker container.

    Security guarantees (per SBOX-01/02/03):
    - CPU and memory limits enforced via Docker API
    - network_mode="none" — no network access
    - No volume mounts — no host filesystem access (SBOX-02)
    - Container killed + removed after execution (SBOX-03)
    - read_only=True on container root filesystem
    """

    def __init__(self) -> None:
        self._client: docker.DockerClient | None = None

    def _get_client(self) -> docker.DockerClient:
        if self._client is None:
            self._client = docker.from_env()
        return self._client

    async def run(
        self,
        code: str,
        lang: str,
        timeout: int | None = None,
    ) -> SandboxResult:
        """Run code in an isolated container and return the result.

        Args:
            code:    Source code to execute.
            lang:    Language identifier — "python" or "bash".
            timeout: Override default policy timeout (seconds).
        """
        policy = get_policy(lang)
        effective_timeout = timeout if timeout is not None else policy.timeout_seconds

        logger.info(
            "sandbox_run_start",
            lang=lang,
            image=policy.image,
            timeout=effective_timeout,
        )

        loop = asyncio.get_event_loop()
        start_ms = time.monotonic()

        try:
            result = await asyncio.wait_for(
                loop.run_in_executor(
                    None,
                    lambda: self._run_sync(code, lang, policy),
                ),
                timeout=effective_timeout,
            )
            result.duration_ms = int((time.monotonic() - start_ms) * 1000)
            return result

        except asyncio.TimeoutError:
            duration_ms = int((time.monotonic() - start_ms) * 1000)
            logger.warning("sandbox_timeout", lang=lang, timeout=effective_timeout)
            # Best-effort cleanup — container may have already exited
            self._cleanup_stale_containers()
            return SandboxResult(
                stdout="",
                stderr=f"Execution timed out after {effective_timeout}s",
                exit_code=-1,
                duration_ms=duration_ms,
                timed_out=True,
            )

    def _run_sync(self, code: str, lang: str, policy: SandboxPolicy) -> SandboxResult:
        """Synchronous inner — runs in executor thread."""
        client = self._get_client()
        container_name = f"{_CONTAINER_NAME_PREFIX}{int(time.monotonic() * 1000)}"

        # Determine run command
        if lang == "python":
            command = ["python", "-c", code]
        else:
            command = ["bash", "-c", code]

        container = None
        try:
            container = client.containers.run(
                image=policy.image,
                command=command,
                name=container_name,
                # SBOX-01: resource limits
                nano_cpus=policy.nano_cpus,
                mem_limit=policy.mem_limit,
                memswap_limit=policy.memswap_limit,
                # SBOX-02: no filesystem access
                network_mode=policy.network_mode,
                read_only=policy.read_only,
                # SBOX-03: auto-remove on exit
                auto_remove=False,   # we remove manually to capture exit code
                detach=True,
            )

            # Wait for container and capture output
            exit_code = container.wait(timeout=60)["StatusCode"]
            raw_logs = container.logs(stdout=True, stderr=True)
            output = raw_logs.decode("utf-8", errors="replace")

            logger.info("sandbox_run_complete", exit_code=exit_code)

            return SandboxResult(
                stdout=output if exit_code == 0 else "",
                stderr=output if exit_code != 0 else "",
                exit_code=exit_code,
                duration_ms=0,  # set by caller
                timed_out=False,
            )

        except Exception as exc:
            logger.error("sandbox_run_error", error=str(exc))
            return SandboxResult(
                stdout="",
                stderr=str(exc),
                exit_code=-1,
                duration_ms=0,
                timed_out=False,
            )
        finally:
            if container is not None:
                try:
                    container.kill()
                except Exception:
                    pass
                try:
                    container.remove(force=True)
                except Exception:
                    pass

    def _cleanup_stale_containers(self) -> None:
        """Remove any blitz-sandbox-* containers that may have leaked."""
        try:
            client = self._get_client()
            containers = client.containers.list(
                filters={"name": _CONTAINER_NAME_PREFIX, "status": "running"}
            )
            for c in containers:
                try:
                    c.kill()
                    c.remove(force=True)
                except Exception:
                    pass
        except Exception as exc:
            logger.warning("sandbox_cleanup_failed", error=str(exc))
```

### Step 6: Create sandbox cleanup Celery task

Create `backend/scheduler/tasks/sandbox_cleanup.py`:

```python
"""
Periodic sandbox cleanup task.

Runs every 5 minutes via Celery Beat. Removes any blitz-sandbox-*
containers that are still running after 5 minutes (leaked containers
from crashes or timeouts that didn't self-clean).

This is belt-and-suspenders cleanup — the executor already calls
container.kill() + container.remove() in its finally block.
"""
import asyncio
import time

import structlog

from scheduler.celery_app import celery_app

logger = structlog.get_logger(__name__)
_MAX_AGE_SECONDS = 300  # 5 minutes


@celery_app.task(name="scheduler.tasks.sandbox_cleanup.cleanup_stale_sandbox_containers")
def cleanup_stale_sandbox_containers() -> dict:
    """Find and remove blitz-sandbox-* containers older than 5 minutes."""
    asyncio.run(_cleanup())
    return {"status": "ok"}


async def _cleanup() -> None:
    try:
        import docker

        client = docker.from_env()
        now = time.time()

        containers = client.containers.list(
            filters={"name": "blitz-sandbox-", "status": "running"}
        )
        removed = 0
        for container in containers:
            try:
                # Container creation time from attrs
                created_ts = container.attrs.get("Created", "")
                # Docker returns ISO 8601; parse to epoch for age check
                import datetime

                created_dt = datetime.datetime.fromisoformat(
                    created_ts.replace("Z", "+00:00")
                )
                age = now - created_dt.timestamp()
                if age > _MAX_AGE_SECONDS:
                    container.kill()
                    container.remove(force=True)
                    removed += 1
                    logger.info("sandbox_cleanup_removed", container=container.name, age_s=int(age))
            except Exception as exc:
                logger.warning("sandbox_cleanup_container_error", error=str(exc))

        logger.info("sandbox_cleanup_complete", removed=removed, checked=len(containers))
    except Exception as exc:
        logger.error("sandbox_cleanup_failed", error=str(exc))
```

"use client";
/**
 * ProviderCard — shows a single OAuth provider with connect/disconnect.
 *
 * Props:
 *   name:         Display name, e.g. "Google Workspace"
 *   provider:     API provider key, e.g. "google"
 *   connectedAt:  ISO timestamp if connected, undefined if not
 *   onDisconnect: callback after successful disconnect
 *   onConnect:    callback to open connect flow
 */
import { useState } from "react";

interface ProviderCardProps {
  name: string;
  provider: string;
  connectedAt?: string;
  onDisconnect: () => void;
  onConnect: () => void;
}

export function ProviderCard({
  name,
  provider,
  connectedAt,
  onDisconnect,
  onConnect,
}: ProviderCardProps) {
  const [confirming, setConfirming] = useState(false);
  const [loading, setLoading] = useState(false);

  const handleDisconnect = async () => {
    if (!confirming) {
      setConfirming(true);
      return;
    }
    setLoading(true);
    setConfirming(false);
    try {
      const res = await fetch(`/api/credentials/${provider}`, { method: "DELETE" });
      if (res.ok || res.status === 204) {
        onDisconnect();
      }
    } finally {
      setLoading(false);
    }
  };

  const handleConnect = async () => {
    setLoading(true);
    try {
      const res = await fetch(`/api/credentials/connect/${provider}`, { method: "POST" });
      const data = (await res.json()) as { url?: string | null; message?: string };
      if (data.url) {
        window.location.href = data.url;
      } else {
        alert(data.message ?? "OAuth not configured. Contact your admin.");
      }
    } finally {
      setLoading(false);
    }
  };

  const isConnected = Boolean(connectedAt);
  const connectedDate = connectedAt
    ? new Date(connectedAt).toLocaleDateString()
    : null;

  return (
    <div className="flex items-center justify-between p-4 border border-gray-200 rounded-lg">
      <div>
        <p className="text-sm font-medium text-gray-900">{name}</p>
        {isConnected ? (
          <p className="text-xs text-green-600 mt-0.5">Connected {connectedDate}</p>
        ) : (
          <p className="text-xs text-gray-400 mt-0.5">Not connected</p>
        )}
      </div>

      <div className="flex items-center gap-2">
        {isConnected ? (
          <button
            onClick={() => void handleDisconnect()}
            disabled={loading}
            className={`text-xs px-3 py-1.5 rounded border transition-colors disabled:opacity-50 ${
              confirming
                ? "border-red-500 text-red-600 hover:bg-red-50"
                : "border-gray-300 text-gray-700 hover:border-gray-400"
            }`}
          >
            {loading ? "..." : confirming ? "Confirm disconnect?" : "Disconnect"}
          </button>
        ) : (
          <button
            onClick={() => void handleConnect()}
            disabled={loading}
            className="text-xs px-3 py-1.5 rounded border border-blue-500 text-blue-600 hover:bg-blue-50 transition-colors disabled:opacity-50"
          >
            {loading ? "..." : "Connect →"}
          </button>
        )}

    </div>
  );

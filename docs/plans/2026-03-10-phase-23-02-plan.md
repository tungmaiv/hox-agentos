```python
@router.post("/fork", status_code=201)
async def fork_skill(
    body: SkillForkRequest,
    user: UserContext = Depends(_require_registry_manager),
    session: AsyncSession = Depends(get_db),
) -> SkillDefinitionResponse:
    """Fork an external skill into the local registry.

    Saves with source_type='imported' and status='pending_review'.
    Runs SecurityScanner and stores the report.
    Admin must approve via the review queue before activation.
    """
    # Check for name collision
    existing = await session.execute(
        select(SkillDefinition).where(SkillDefinition.name == body.name)
    )
    if existing.scalar_one_or_none() is not None:
        raise HTTPException(
            status_code=409,
            detail=f"A skill named '{body.name}' already exists in the registry",
        )

    # Security scan
    scanner = SecurityScanner()
    skill_data: dict[str, Any] = {
        "name": body.name,
        "skill_type": body.skill_type,
        "source_type": "imported",
        "instruction_markdown": body.instruction_markdown,
        "procedure_json": body.procedure_json,
    }
    report = scanner.scan(skill_data, source_url=body.source_url)

    skill = SkillDefinition(
        name=body.name,
        display_name=body.display_name,
        description=body.description,
        version=body.version,
        skill_type=body.skill_type,
        source_type="imported",
        status="pending_review",
        instruction_markdown=body.instruction_markdown,
        procedure_json=body.procedure_json,
        input_schema=body.input_schema,
        output_schema=body.output_schema,
        source_url=body.source_url,
        license=body.license,
        tags=body.tags,
        category=body.category,
        security_score=report.score,
        security_report={
            "score": report.score,
            "factors": report.factors,
            "recommendation": report.recommendation,
            "injection_matches": report.injection_matches,
        },
        created_by=user["user_id"],
    )
    session.add(skill)
    await session.commit()
    await session.refresh(skill)

    logger.info(
        "skill_forked",
        skill_id=str(skill.id),
        name=skill.name,
        score=report.score,
        recommendation=report.recommendation,
        user_id=str(user["user_id"]),
    )
    return SkillDefinitionResponse.model_validate(skill)
```

### Step 1: Create the component

```tsx
// frontend/src/components/admin/similar-skills-panel.tsx
"use client";
/**
 * SimilarSkillsPanel — shows top external skills matching the current query.
 *
 * Debounces search on name/description changes.
 * Two actions per skill: "Fork" (creates a local copy) and "Reference" (injects into chat).
 */
import { useCallback, useEffect, useState } from "react";

interface SimilarSkill {
  name: string;
  description: string | null;
  version: string | null;
  repository_name: string;
  repository_id: string;
  category: string | null;
  tags: string[] | null;
  license: string | null;
  author: string | null;
  // Full skill definition for fork/reference
  skill_type?: string;
  instruction_markdown?: string;
  procedure_json?: unknown;
  source_url?: string;
}

interface SimilarSkillsPanelProps {
  query: string;
  /** Called when user forks a skill — receives the new skill id. */
  onFork: (skillId: string, skillName: string) => void;
  /** Called when user uses a skill as reference — receives instruction text. */
  onReference: (content: string, skillName: string) => void;
}

function SkillCard({
  skill,
  onFork,
  onReference,
}: {
  skill: SimilarSkill;
  onFork: () => void;
  onReference: () => void;
}) {
  const [forking, setForking] = useState(false);
  const [forkError, setForkError] = useState<string | null>(null);

  const handleFork = useCallback(async () => {
    setForking(true);
    setForkError(null);
    try {
      onFork();
    } finally {
      setForking(false);
    }
  }, [onFork]);

  return (
    <div className="border border-gray-200 rounded-md p-3 bg-white hover:border-gray-300 transition-colors">
      <div className="flex items-start justify-between gap-2">
        <div className="flex-1 min-w-0">
          <p className="text-xs font-semibold text-gray-900 truncate">{skill.name}</p>
          {skill.description && (
            <p className="text-xs text-gray-500 mt-0.5 line-clamp-2">{skill.description}</p>
          )}
          <p className="text-xs text-gray-400 mt-1">
            {skill.repository_name}
            {skill.version ? ` · v${skill.version}` : ""}
            {skill.license ? ` · ${skill.license}` : ""}
          </p>
        </div>
      </div>
      {forkError && (
        <p className="text-xs text-red-600 mt-1">{forkError}</p>
      )}
      <div className="flex gap-1.5 mt-2">
        <button
          type="button"
          onClick={handleFork}
          disabled={forking}
          className="px-2 py-1 text-xs bg-blue-600 text-white rounded hover:bg-blue-700 disabled:opacity-50 font-medium"
        >
          {forking ? "Forking..." : "Fork this"}
        </button>
        <button
          type="button"
          onClick={onReference}
          className="px-2 py-1 text-xs bg-gray-100 text-gray-700 rounded hover:bg-gray-200 font-medium"
        >
          Use as reference
        </button>
      </div>
    </div>
  );
}

export function SimilarSkillsPanel({
  query,
  onFork,
  onReference,
}: SimilarSkillsPanelProps) {
  const [skills, setSkills] = useState<SimilarSkill[]>([]);
  const [loading, setLoading] = useState(false);

  // Debounced search — fires 600ms after query stabilises
  useEffect(() => {
    if (query.trim().length < 3) {
      setSkills([]);
      return;
    }

    const id = setTimeout(async () => {
      setLoading(true);
      try {
        const res = await fetch(
          `/api/admin/skills/similar?q=${encodeURIComponent(query)}&limit=5`
        );
        if (!res.ok) return;
        const data = (await res.json()) as SimilarSkill[];
        setSkills(data);
      } catch {
        // Silently ignore network errors — panel is a discovery aid, not critical
      } finally {
        setLoading(false);
      }
    }, 600);

    return () => clearTimeout(id);
  }, [query]);

  if (query.trim().length < 3) return null;

  return (
    <div className="border border-gray-200 rounded-lg overflow-hidden bg-gray-50">
      <div className="px-3 py-2 border-b border-gray-200 flex items-center justify-between">
        <p className="text-xs font-semibold text-gray-700">Similar Skills</p>
        {loading && (
          <span className="text-xs text-gray-400">Searching...</span>
        )}
      </div>
      <div className="p-3">
        {!loading && skills.length === 0 && (
          <p className="text-xs text-gray-400 text-center py-2">
            No similar skills found in external repositories
          </p>
        )}
        <div className="space-y-2">
          {skills.map((skill) => (
            <SkillCard
              key={`${skill.repository_id}-${skill.name}`}
              skill={skill}
              onFork={async () => {
                const res = await fetch("/api/admin/skills/fork", {
                  method: "POST",
                  headers: { "Content-Type": "application/json" },
                  body: JSON.stringify({
                    name: skill.name,
                    description: skill.description,
                    skill_type: skill.skill_type ?? "instructional",
                    instruction_markdown: skill.instruction_markdown,
                    procedure_json: skill.procedure_json,
                    source_url: skill.source_url,
                    license: skill.license,
                    tags: skill.tags,
                    category: skill.category,
                  }),
                });
                if (!res.ok) {
                  const body = (await res.json().catch(() => ({}))) as Record<string, unknown>;
                  const detail = typeof body.detail === "string" ? body.detail : `HTTP ${res.status}`;
                  throw new Error(detail);
                }
                const saved = (await res.json()) as { id: string };
                onFork(saved.id, skill.name);
              }}
              onReference={() => {
                const content =
                  skill.instruction_markdown ??
                  (skill.procedure_json
                    ? `Procedure from ${skill.name}:\n\`\`\`json\n${JSON.stringify(skill.procedure_json, null, 2)}\n\`\`\``
                    : `Skill: ${skill.name}\n${skill.description ?? ""}`);
                onReference(content, skill.name);
              }}
```

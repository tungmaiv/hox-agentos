async def scan_preview(
    body: ScanPreviewRequest,
    user: UserContext = Depends(_require_registry_manager),
) -> SecurityReportResponse:
    """Run SecurityScanner on a skill definition without saving.

    Returns trust score, factor breakdown, and recommendation.
    Does not write to the database.
    """
    scanner = SecurityScanner()
    skill_data: dict[str, Any] = {
        "name": body.name,
        "skill_type": body.skill_type,
        "source_type": body.source_type,
        "instruction_markdown": body.instruction_markdown,
        "procedure_json": body.procedure_json,
    }
    report = scanner.scan(skill_data, source_url=body.source_url)
    logger.info(
        "skill_scan_preview",
        skill_name=body.name,
        score=report.score,
        recommendation=report.recommendation,
        user_id=str(user["user_id"]),
    )
    return SecurityReportResponse(
        score=report.score,
        factors=report.factors,
        recommendation=report.recommendation,
        injection_matches=report.injection_matches,
    )

@router.post("", status_code=201)
async def create_skill(
    body: SkillDefinitionCreate,
    user: UserContext = Depends(_require_registry_manager),
    session: AsyncSession = Depends(get_db),
) -> SkillDefinitionResponse:
    """Create a new skill definition.

    For user_created skills: runs SecurityScanner and sets status=pending_review
    if score < 80. Builtin and imported skills skip this gate (they have their
    own scan path on import).
    """
    skill = SkillDefinition(
        name=body.name,
        display_name=body.display_name,
        description=body.description,
        version=body.version,
        skill_type=body.skill_type,
        slash_command=body.slash_command,
        source_type=body.source_type,
        instruction_markdown=body.instruction_markdown,
        procedure_json=body.procedure_json,
        input_schema=body.input_schema,
        output_schema=body.output_schema,
        license=body.license,
        compatibility=body.compatibility,
        metadata_json=body.metadata_json,
        allowed_tools=body.allowed_tools,
        tags=body.tags,
        category=body.category,
        source_url=body.source_url,
        created_by=user["user_id"],
    )

    # Security gate: scan user_created skills before activation
    if body.source_type == "user_created":
        scanner = SecurityScanner()
        skill_data: dict[str, Any] = {
            "name": body.name,
            "skill_type": body.skill_type,
            "source_type": body.source_type,
            "instruction_markdown": body.instruction_markdown,
            "procedure_json": body.procedure_json,
        }
        report = scanner.scan(skill_data, source_url=body.source_url)
        skill.security_score = report.score
        skill.security_report = {
            "score": report.score,
            "factors": report.factors,
            "recommendation": report.recommendation,
            "injection_matches": report.injection_matches,
        }
        if report.score < 80:
            skill.status = "pending_review"
            logger.info(
                "skill_quarantined_on_create",
                skill_name=body.name,
                score=report.score,
                recommendation=report.recommendation,
                user_id=str(user["user_id"]),
            )

    session.add(skill)
    await session.commit()
    await session.refresh(skill)
    logger.info(
        "admin_skill_created",
        skill_id=str(skill.id),
        name=skill.name,
        status=skill.status,
        user_id=str(user["user_id"]),
    )
    return SkillDefinitionResponse.model_validate(skill)
```

// frontend/src/components/admin/security-report-card.tsx
"use client";
/**
 * SecurityReportCard — displays a skill's security scan result.
 *
 * Shows trust score (0-100), factor breakdown, injection warnings,
 * and a recommendation badge (approve / review / reject).
 */

interface SecurityReport {
  score: number;
  factors: Record<string, number>;
  recommendation: "approve" | "review" | "reject";
  injection_matches?: string[];
}

interface SecurityReportCardProps {
  report: SecurityReport;
  /** Called when user confirms saving (despite warnings). */
  onConfirm?: () => void;
  /** Called when user cancels. */
  onCancel?: () => void;
  /** True while the parent is saving. */
  saving?: boolean;
}

const FACTOR_LABELS: Record<string, string> = {
  source_reputation: "Source reputation",
  tool_scope: "Tool scope",
  prompt_safety: "Prompt safety",
  complexity: "Complexity",
  dependency_risk: "Dependency risk",
  data_flow_risk: "Data flow risk",
};

function ScoreBadge({ score }: { score: number }) {
  const color =
    score >= 80
      ? "bg-green-100 text-green-800 border-green-200"
      : score >= 60
        ? "bg-amber-100 text-amber-800 border-amber-200"
        : "bg-red-100 text-red-800 border-red-200";
  return (
    <span
      className={`inline-flex items-center px-3 py-1 rounded-full text-sm font-semibold border ${color}`}
    >
      {score}/100
    </span>
  );
}

function RecommendationBadge({
  recommendation,
}: {
  recommendation: SecurityReport["recommendation"];
}) {
  const styles = {
    approve: "bg-green-100 text-green-800 border-green-200",
    review: "bg-amber-100 text-amber-800 border-amber-200",
    reject: "bg-red-100 text-red-800 border-red-200",
  };
  const labels = {
    approve: "Approve",
    review: "Needs review",
    reject: "Reject",
  };
  return (
    <span
      className={`inline-flex items-center px-2.5 py-0.5 rounded text-xs font-medium border ${styles[recommendation]}`}
    >
      {labels[recommendation]}
    </span>
  );
}

function FactorBar({ score }: { score: number }) {
  const color =
    score >= 80 ? "bg-green-500" : score >= 60 ? "bg-amber-500" : "bg-red-500";
  return (
    <div className="flex items-center gap-2">
      <div className="flex-1 bg-gray-100 rounded-full h-1.5">
        <div
          className={`h-1.5 rounded-full ${color}`}
          style={{ width: `${score}%` }}
        />
      </div>
      <span className="text-xs text-gray-500 w-8 text-right">{score}</span>
    </div>
  );
}

export function SecurityReportCard({
  report,
  onConfirm,
  onCancel,
  saving = false,
}: SecurityReportCardProps) {
  const isBlocked = report.recommendation === "reject";
  const needsReview = report.recommendation === "review";

  return (
    <div className="border border-gray-200 rounded-lg overflow-hidden bg-white">
      {/* Header */}
      <div className="px-4 py-3 border-b border-gray-200 bg-gray-50 flex items-center justify-between">
        <h3 className="text-sm font-semibold text-gray-900">Security Scan Result</h3>
        <div className="flex items-center gap-2">
          <ScoreBadge score={report.score} />
          <RecommendationBadge recommendation={report.recommendation} />
        </div>
      </div>

      <div className="p-4 space-y-4">
        {/* Factor breakdown */}
        <div>
          <p className="text-xs font-medium text-gray-500 uppercase tracking-wide mb-2">
            Factor Breakdown
          </p>
          <div className="space-y-1.5">
            {Object.entries(report.factors).map(([key, value]) => (
              <div key={key} className="grid grid-cols-[1fr_auto] gap-2 items-center">
                <span className="text-xs text-gray-700">
                  {FACTOR_LABELS[key] ?? key}
                </span>
                <div className="w-32">
                  <FactorBar score={value} />
                </div>
              </div>
            ))}
          </div>
        </div>

        {/* Injection warnings */}
        {report.injection_matches && report.injection_matches.length > 0 && (
          <div className="p-3 bg-red-50 border border-red-200 rounded-md">
            <p className="text-xs font-medium text-red-800 mb-1">
              Prompt injection patterns detected:
            </p>
            <ul className="space-y-0.5">
              {report.injection_matches.map((match, i) => (
                <li key={i} className="text-xs text-red-700 font-mono">
                  &quot;{match}&quot;
                </li>
              ))}
            </ul>
          </div>
        )}

        {/* Status message */}
        {report.recommendation === "approve" && (
          <p className="text-xs text-green-700 bg-green-50 border border-green-200 rounded p-2">
            This skill passed the security review and will be activated immediately.
          </p>
        )}
        {needsReview && (
          <p className="text-xs text-amber-700 bg-amber-50 border border-amber-200 rounded p-2">
            This skill will be saved as <strong>pending review</strong>. An admin must approve it
            before it becomes active.
          </p>
        )}
        {isBlocked && (
          <p className="text-xs text-red-700 bg-red-50 border border-red-200 rounded p-2">
            This skill was flagged as high risk and will require admin review before activation.
          </p>
        )}

        {/* Actions */}
        {(onConfirm || onCancel) && (
          <div className="flex items-center justify-end gap-2 pt-2 border-t border-gray-100">
            {onCancel && (
              <button
                type="button"
                onClick={onCancel}
                className="px-3 py-1.5 text-xs text-gray-700 bg-white border border-gray-300 rounded-md hover:bg-gray-50"
              >
                Cancel
              </button>
            )}
            {onConfirm && (
              <button
                type="button"
                onClick={onConfirm}
                disabled={saving}
                className={`px-3 py-1.5 text-xs text-white rounded-md font-medium disabled:opacity-50 ${
                  isBlocked || needsReview
                    ? "bg-amber-600 hover:bg-amber-700"
                    : "bg-blue-600 hover:bg-blue-700"
                }`}
              >
                {saving
                  ? "Saving..."
                  : isBlocked || needsReview
                    ? "Save for Review"
                    : "Save to Registry"}
              </button>
            )}
          </div>
        )}
      </div>
    </div>
  );
}
```

### Step 3: Replace `handleSubmit` with scan-first flow

Replace the existing `handleSubmit` with a two-phase version:

```tsx
// Phase 1: scan (for skills only), then show report
const handleSubmit = useCallback(async () => {
  if (!artifactType || !formState.name) return;

  // Non-skill artifacts save directly (no scan gate)
  if (artifactType !== "skill") {
    await _doSave();
    return;
  }

  // For skills: scan first
  setIsScanning(true);
  try {
    const scanPayload = {
      name: formState.name,
      skill_type: "instructional" as const,
      source_type: "user_created" as const,
      instruction_markdown: formState.instruction_markdown || undefined,
    };
    const res = await fetch("/api/admin/skills/scan-preview", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(scanPayload),
    });
    if (!res.ok) throw new Error(`Scan failed: HTTP ${res.status}`);
    const report = await res.json() as {
      score: number;
      factors: Record<string, number>;
      recommendation: "approve" | "review" | "reject";
      injection_matches?: string[];
    };
    setScanReport(report);
    setShowScanModal(true);
  } catch (err) {
    toast.error(err instanceof Error ? err.message : "Security scan failed");
  } finally {
    setIsScanning(false);
  }
}, [artifactType, formState]);

// Phase 2: confirmed save after viewing scan report
const _doSave = useCallback(async () => {
  if (!artifactType || !formState.name) return;

  const path = TYPE_TO_PATH[artifactType];
  if (!path) {
    toast.error("Unknown artifact type");
    return;
  }

  setIsSubmitting(true);
  setShowScanModal(false);

  const payload: Record<string, unknown> = {
    name: formState.name,
    description: formState.description || undefined,
    version: formState.version || "1.0.0",
  };

  switch (artifactType) {
    case "agent":
      payload.config_json = {
        model_alias: formState.model_alias || undefined,
        system_prompt: formState.system_prompt || undefined,
      };
      break;
    case "tool":
      payload.handler_module = formState.handler_module || undefined;
      payload.sandbox_required = formState.sandbox_required;
      payload.required_permissions = formState.required_permissions;
      break;
    case "skill":
      payload.skill_type = "instructional";
      payload.source_type = "user_created";
      payload.instruction_markdown = formState.instruction_markdown || undefined;
      payload.entry_point = formState.entry_point || undefined;
      payload.required_permissions = formState.required_permissions;
      payload.sandbox_required = formState.sandbox_required;
      break;
    case "mcp_server":
      payload.url = formState.url;
      payload.auth_token = formState.auth_token || undefined;
      break;
  }

  try {
    const res = await fetch(`/api/admin/${path}`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(payload),
    });

    if (!res.ok) {
      const body = (await res.json().catch(() => ({}))) as Record<string, unknown>;
      let errorMsg = `HTTP ${res.status}`;
      if (typeof body.detail === "string") {
        errorMsg = body.detail;
      } else if (Array.isArray(body.detail)) {
        errorMsg = (body.detail as Array<Record<string, unknown>>)
          .map((e) => {
            const loc = Array.isArray(e.loc) ? (e.loc as string[]).join(" > ") : "";
            return loc ? `${loc}: ${e.msg}` : String(e.msg ?? e);
          })
          .join("; ");
      }
      throw new Error(errorMsg);
    }

    const label = TYPE_TO_LABEL[artifactType] ?? artifactType;
    const tabPath = path;
    const savedBody = await res.json() as { status?: string };
    if (savedBody.status === "pending_review") {
      toast.info(`${label} saved — pending admin review before activation`);
    } else {
      toast.success(`${label} created — view in /${tabPath} tab`);
    }

    setFormState(EMPTY_FORM);
    setArtifactType(null);
    setAiFilledFields(new Set());
    setScanReport(null);
  } catch (err) {
    toast.error(
      err instanceof Error ? err.message : "Failed to create artifact — try again"
    );
  } finally {
    setIsSubmitting(false);
  }
}, [artifactType, formState]);
```

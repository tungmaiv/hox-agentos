# Phase 2 — Agent Core and Conversational Chat: Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Wire a streaming LangGraph ReAct agent behind the AG-UI protocol, with full chat UI, per-user conversation persistence, and an AES-256 credential store stub.

**Architecture:** LiteLLM proxy routes all LLM calls. A LangGraph `StateGraph` with a single `master_agent` node handles chat. CopilotKit connects the Next.js frontend to the FastAPI backend via a JWT-authenticated Next.js proxy route. Memory is loaded from `memory_conversations` before graph invocation and saved after. Conversation ID is a UUID generated by the frontend.

**Tech Stack:** LangGraph 0.2+, CopilotKit (Python SDK + React), `@copilotkit/react-ui`, `cryptography` (AES-256-GCM), SQLAlchemy async ORM, Alembic, aiosqlite (tests)

**Step 4: Create `agents/master_agent.py`**

```python
# backend/agents/master_agent.py
"""
Master agent — Phase 2 conversational ReAct graph.

Single master_agent node: receives messages, calls blitz/master LLM,
returns response. No tools in Phase 2 — tool nodes added in Phase 3.

Evolution path:
  Phase 2: master_agent node → END  (conversational only)
  Phase 3: add sub-agent nodes + conditional routing edges
  Phase 4: canvas workflows compile to separate StateGraphs (independent)
"""
from langchain_core.messages import BaseMessage
from langgraph.graph import END, StateGraph
from langgraph.graph.state import CompiledStateGraph

from agents.state.types import BlitzState
from core.config import get_llm

import structlog

logger = structlog.get_logger(__name__)


async def _master_node(state: BlitzState) -> dict[str, list[BaseMessage]]:
    """Call blitz/master LLM with current messages and return response."""
    llm = get_llm("blitz/master")
    response = await llm.ainvoke(state["messages"])
    logger.info("master_agent_response", response_type=type(response).__name__)
    return {"messages": [response]}


def create_master_graph() -> CompiledStateGraph:
    """
    Build and compile the master agent StateGraph.

    Returns a compiled graph ready for CopilotKit or direct invocation.
    Call once at startup and reuse — compilation is expensive.
    """
    graph = StateGraph(BlitzState)
    graph.add_node("master_agent", _master_node)
    graph.set_entry_point("master_agent")
    graph.add_edge("master_agent", END)
    return graph.compile()
```

```python
# backend/gateway/runtime.py
"""
CopilotKit runtime — wraps the master agent graph for AG-UI streaming.

The /api/copilotkit endpoint:
  1. Gate 1+2+3 security applied via FastAPI dependency
  2. Extracts thread_id (conversation_id) from CopilotKit request body
  3. Passes user context to the graph via Python contextvars
  4. Streams back tokens via CopilotKit's event-source protocol
"""
import contextvars
from uuid import UUID

from copilotkit import CopilotKitSDK, LangGraphAgent
from fastapi import APIRouter, Depends, Request
from fastapi.responses import Response

from agents.master_agent import create_master_graph
from core.models.user import UserContext
from security.acl import check_tool_acl, log_tool_call
from security.deps import get_current_user
from security.rbac import has_permission

import structlog
import time

logger = structlog.get_logger(__name__)

# Context var — allows graph nodes to access the current user without
# threading it through every function signature.
current_user_ctx: contextvars.ContextVar[UserContext] = contextvars.ContextVar(
    "current_user_ctx"
)

# Build the graph once at module load — compilation is expensive.
_master_graph = create_master_graph()

_sdk = CopilotKitSDK(
    agents=[
        LangGraphAgent(
            name="blitz_master",
            description="Blitz AgentOS master conversational agent",
            graph=_master_graph,
        )
    ]
)

router = APIRouter(prefix="/api", tags=["copilotkit"])


@router.post(
    "/copilotkit",
    responses={
        401: {"description": "Missing or invalid JWT"},
        403: {"description": "Insufficient permissions"},
    },
)
async def copilotkit_endpoint(
    request: Request,
    user: UserContext = Depends(get_current_user),
) -> Response:
    """
    AG-UI streaming endpoint for CopilotKit.

    Security: same 3-gate chain as /api/agents/chat.
    The JWT Authorization header is injected by the Next.js proxy route
    (frontend/src/app/api/copilotkit/route.ts) from the server-side session.
    """
    from sqlalchemy.ext.asyncio import AsyncSession
    from core.db import async_session
    from fastapi import HTTPException

    start_ms = int(time.monotonic() * 1000)

    # Gate 2: RBAC
    if not has_permission(user, "chat"):
        elapsed = int(time.monotonic() * 1000) - start_ms
        await _log_and_deny(user, elapsed)
        raise HTTPException(status_code=403, detail="Permission denied")

    # Gate 3: Tool ACL
    async with async_session() as session:
        allowed = await check_tool_acl(user["user_id"], "agents.chat", session)
    elapsed = int(time.monotonic() * 1000) - start_ms
    await log_tool_call(user["user_id"], "agents.chat", allowed, elapsed)

    if not allowed:
        raise HTTPException(status_code=403, detail="Permission denied by ACL")

    # Set user context for graph nodes to access via current_user_ctx.get()
    token = current_user_ctx.set(user)
    try:
        body = await request.json()
        return await _sdk.process_messages(body=body)
    finally:
        current_user_ctx.reset(token)


async def _log_and_deny(user: UserContext, elapsed: int) -> None:
    await log_tool_call(user["user_id"], "agents.chat", False, elapsed)
```

**Step 1: Create the proxy route**

This route injects the server-side JWT into every CopilotKit request forwarded to the FastAPI backend. The JWT never touches the browser.

```typescript
// frontend/src/app/api/copilotkit/route.ts

/**
 * CopilotKit proxy — forwards AG-UI requests to FastAPI with JWT.
 *
 * Security model:
 * - Reads access token from next-auth server-side session (never from browser)
 * - Injects Authorization: Bearer header before forwarding to backend
 * - Token never appears in browser network tab (server-to-server request)
 *
 * The CopilotKit frontend provider points to this route (/api/copilotkit).
 * This route transparently proxies to http://backend:8000/api/copilotkit
 * (or localhost:8000 in local dev).
 */

import { auth } from "@/auth";

const BACKEND_URL =
  process.env.BACKEND_URL ?? process.env.NEXT_PUBLIC_API_URL ?? "http://localhost:8000";

export async function POST(req: Request): Promise<Response> {
  const session = await auth();

  const headers: Record<string, string> = {
    "Content-Type": "application/json",
  };

  const token = (session as { accessToken?: string } | null)?.accessToken;
  if (token) {
    headers["Authorization"] = `Bearer ${token}`;
  }

  // Forward the CopilotKit AG-UI request body to FastAPI
  const body = await req.text();

  const backendResponse = await fetch(`${BACKEND_URL}/api/copilotkit`, {
    method: "POST",
    headers,
    body,
    // Required for streaming: don't buffer the response body
    // @ts-expect-error - duplex is a valid fetch option but not in all TS types
    duplex: "half",
  });

  // Stream the response back to the frontend without buffering
  return new Response(backendResponse.body, {
    status: backendResponse.status,
    headers: Object.fromEntries(backendResponse.headers.entries()),
  });
}
```

```typescript
// frontend/src/hooks/use-chat.ts
"use client";

/**
 * useChat — manages conversation ID for the current chat session.
 *
 * The conversation ID is a UUID generated by the frontend.
 * It's passed to the backend as the CopilotKit threadId.
 * A new ID starts a fresh conversation (backend has no history for that ID).
 */
import { useState, useCallback } from "react";

interface UseChatReturn {
  conversationId: string;
  startNewConversation: () => void;
}

export function useChat(): UseChatReturn {
  const [conversationId, setConversationId] = useState<string>(
    () => crypto.randomUUID()
  );

  const startNewConversation = useCallback(() => {
    setConversationId(crypto.randomUUID());
  }, []);

  return { conversationId, startNewConversation };
}
```

```python
# backend/memory/short_term.py
"""
Short-term memory (Tier 1) — conversation turn storage and retrieval.

All queries are parameterized on user_id from JWT.
Caller is responsible for passing a valid session from Depends(get_db).
"""
from uuid import UUID

from langchain_core.messages import AIMessage, BaseMessage, HumanMessage
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from core.models.conversation import ConversationTurn

import structlog

logger = structlog.get_logger(__name__)

_ROLE_TO_MESSAGE: dict[str, type[BaseMessage]] = {
    "user": HumanMessage,
    "assistant": AIMessage,
    # "tool" messages handled as HumanMessage for simplicity in Phase 2
    "tool": HumanMessage,
}


async def load_recent_turns(
    user_id: UUID,
    conversation_id: UUID,
    *,
    n: int = 20,
    session: AsyncSession,
) -> list[BaseMessage]:
    """
    Load the last n turns for a user's conversation as LangChain messages.

    Returns an empty list if no turns exist (new conversation).
    user_id isolation: WHERE user_id = $1 ensures User A cannot read User B's turns.
    """
    result = await session.execute(
        select(ConversationTurn)
        .where(
            ConversationTurn.user_id == user_id,
            ConversationTurn.conversation_id == conversation_id,
        )
        .order_by(ConversationTurn.created_at.asc())
        .limit(n)
    )
    turns = result.scalars().all()
    messages: list[BaseMessage] = []
    for turn in turns:
        msg_class = _ROLE_TO_MESSAGE.get(turn.role, HumanMessage)
        messages.append(msg_class(content=turn.content))
    logger.debug(
        "memory_loaded",
        user_id=str(user_id),
        conversation_id=str(conversation_id),
        count=len(messages),
    )
    return messages


async def save_turn(
    user_id: UUID,
    conversation_id: UUID,
    role: str,
    content: str,
    *,
    session: AsyncSession,
) -> None:
    """
    Save one conversation turn to memory_conversations.

    role must be one of: "user", "assistant", "tool"
    """
    turn = ConversationTurn(
        user_id=user_id,
        conversation_id=conversation_id,
        role=role,
        content=content,
    )
    session.add(turn)
    await session.commit()
    logger.debug(
        "memory_saved",
        user_id=str(user_id),
        conversation_id=str(conversation_id),
        role=role,
    )
```

```python
        async with async_session() as session:

        # Prepend history to the messages in the CopilotKit body

            history_dicts = [

                 "content": m.content}

            body["messages"] = history_dicts + body["messages"]

        response = await _sdk.process_messages(body=body)

        # Save the latest user message and response

        try:
            current_messages = body.get("messages", [])
            if current_messages:
                last_user = next(
                    (m for m in reversed(current_messages) if m.get("role") == "user"),

                            session=session,
```

```python
# backend/api/routes/conversations.py
"""
Conversation list endpoint — powers the chat sidebar.

Returns a summary of each conversation (id + first message snippet)
for the current user. All queries parameterized on user_id from JWT.
"""
from uuid import UUID

from fastapi import APIRouter, Depends
from pydantic import BaseModel
from sqlalchemy import distinct, select
from sqlalchemy.ext.asyncio import AsyncSession

from core.db import get_db
from core.models.conversation import ConversationTurn
from core.models.user import UserContext
from security.deps import get_current_user

router = APIRouter(prefix="/conversations", tags=["conversations"])


class ConversationSummary(BaseModel):
    conversation_id: str
    first_message: str
    message_count: int


@router.get("", response_model=list[ConversationSummary])
async def list_conversations(
    user: UserContext = Depends(get_current_user),
    session: AsyncSession = Depends(get_db),
) -> list[ConversationSummary]:
    """
    List all conversations for the current user, most recent first.
    Returns up to 50 conversations with a snippet of the first message.
    """
    # Get distinct conversation IDs for this user
    result = await session.execute(
        select(ConversationTurn)
        .where(ConversationTurn.user_id == user["user_id"])
        .order_by(ConversationTurn.created_at.asc())
    )
    turns = result.scalars().all()

    # Group by conversation_id — preserve insertion order of first appearance
    seen: dict[str, ConversationSummary] = {}
    counts: dict[str, int] = {}
    for turn in turns:
        cid = str(turn.conversation_id)
        if cid not in seen:
            seen[cid] = ConversationSummary(
                conversation_id=cid,
                first_message=turn.content[:80],
                message_count=0,
            )
        counts[cid] = counts.get(cid, 0) + 1

    for cid, summary in seen.items():
        summary.message_count = counts[cid]

    # Return most recent conversations first (reverse insertion order)
    summaries = list(reversed(list(seen.values())))
    return summaries[:50]
```

```python
# backend/security/credentials.py
"""
Credential store — AES-256-GCM encrypted OAuth token storage.

Tokens are NEVER stored in plaintext and NEVER logged.
The encryption key is derived from settings.secret_key (32 bytes via SHA-256).
A fresh 12-byte IV (nonce) is generated for every encryption call.

Phase 2: encrypt/decrypt + CRUD. OAuth flows wired in Phase 3.
"""
import hashlib
from datetime import datetime
from uuid import UUID

from cryptography.hazmat.primitives.ciphers.aead import AESGCM
from sqlalchemy import delete, select
from sqlalchemy.ext.asyncio import AsyncSession

from core.config import settings
from core.models.credential import UserCredential

import os
import structlog

logger = structlog.get_logger(__name__)


def _derive_key() -> bytes:
    """Derive a 32-byte AES key from settings.secret_key using SHA-256."""
    return hashlib.sha256(settings.secret_key.encode()).digest()


def encrypt_token(token: str) -> tuple[bytes, bytes]:
    """
    Encrypt a plaintext token using AES-256-GCM.

    Returns (ciphertext, iv). Both must be stored; both required for decryption.
    A fresh random IV is generated on every call — same token → different ciphertext.
    """
    key = _derive_key()
    iv = os.urandom(12)  # 96-bit nonce for GCM
    aesgcm = AESGCM(key)
    ciphertext = aesgcm.encrypt(iv, token.encode(), None)
    return ciphertext, iv


def decrypt_token(ciphertext: bytes, iv: bytes) -> str:
    """
    Decrypt an AES-256-GCM ciphertext.

    Raises cryptography.exceptions.InvalidTag if tampered.
    """
    key = _derive_key()
    aesgcm = AESGCM(key)
    plaintext = aesgcm.decrypt(iv, ciphertext, None)
    return plaintext.decode()


async def store_credential(
    user_id: UUID,
    provider: str,
    token: str,
    *,
    expires_at: datetime | None = None,
    session: AsyncSession,
) -> None:
    """
    Encrypt and store an OAuth token for a user+provider.

    If a credential already exists for this user+provider, it is replaced.
    The plaintext token is never logged or persisted.
    """
    ciphertext, iv = encrypt_token(token)

    # Upsert: delete existing then insert fresh
    await session.execute(
        delete(UserCredential).where(
            UserCredential.user_id == user_id,
            UserCredential.provider == provider,
        )
    )
    credential = UserCredential(
        user_id=user_id,
        provider=provider,
        encrypted_token=ciphertext,
        iv=iv,
        expires_at=expires_at,
    )
    session.add(credential)
    await session.commit()
    logger.info("credential_stored", user_id=str(user_id), provider=provider)


async def get_credential(
    user_id: UUID,
    provider: str,
    *,
    session: AsyncSession,
) -> str | None:
    """
    Retrieve and decrypt a stored OAuth token.

    Returns None if no credential exists for this user+provider.
    The decrypted token is returned in memory only — never logged.
    """
    result = await session.execute(
        select(UserCredential).where(
            UserCredential.user_id == user_id,
            UserCredential.provider == provider,
        )
    )
    cred = result.scalar_one_or_none()
    if cred is None:
        return None
    return decrypt_token(cred.encrypted_token, cred.iv)


async def delete_credential(
    user_id: UUID,
    provider: str,
    *,
    session: AsyncSession,
) -> None:
    """Delete a stored credential for a user+provider."""
    await session.execute(
        delete(UserCredential).where(
            UserCredential.user_id == user_id,
            UserCredential.provider == provider,
        )
    )
    await session.commit()
    logger.info("credential_deleted", user_id=str(user_id), provider=provider)
```

```python
# backend/api/routes/credentials.py
"""
Credential management endpoints — Phase 2 stub.

GET  /api/credentials               — list provider names (not tokens) for current user
DELETE /api/credentials/{provider}  — remove a credential

POST is intentionally absent — OAuth flows are wired in Phase 3.
Tokens are NEVER returned in API responses.
"""
from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from core.db import get_db
from core.models.credential import UserCredential
from core.models.user import UserContext
from security.credentials import delete_credential
from security.deps import get_current_user

router = APIRouter(prefix="/credentials", tags=["credentials"])


class CredentialInfo(BaseModel):
    provider: str
    expires_at: str | None


@router.get("", response_model=list[CredentialInfo])
async def list_credentials(
    user: UserContext = Depends(get_current_user),
    session: AsyncSession = Depends(get_db),
) -> list[CredentialInfo]:
    """
    List provider names for the current user's connected accounts.
    Tokens are NEVER included in the response.
    """
    result = await session.execute(
        select(UserCredential).where(UserCredential.user_id == user["user_id"])
    )
    creds = result.scalars().all()
    return [
        CredentialInfo(
            provider=c.provider,
            expires_at=c.expires_at.isoformat() if c.expires_at else None,
        )
        for c in creds
    ]


@router.delete("/{provider}", status_code=204)
async def remove_credential(
    provider: str,
    user: UserContext = Depends(get_current_user),
    session: AsyncSession = Depends(get_db),
) -> None:
    """Remove a stored credential for the given provider."""
    # Verify it exists before deleting
    result = await session.execute(
        select(UserCredential).where(
            UserCredential.user_id == user["user_id"],
            UserCredential.provider == provider,
        )
    )
    if result.scalar_one_or_none() is None:
        raise HTTPException(status_code=404, detail=f"No credential for provider '{provider}'")
    await delete_credential(user["user_id"], provider, session=session)
```

    ```python
    from alembic import op
    import sqlalchemy as sa
    import json, pathlib, uuid

    def upgrade():
        bind = op.get_bind()
        fixtures_dir = pathlib.Path(__file__).parent.parent.parent / "data" / "workflow_templates"
        for filename in ["morning_digest.json", "alert.json"]:
            data = json.loads((fixtures_dir / filename).read_text())
            name = filename.replace(".json", "").replace("_", " ").title()
            bind.execute(sa.text(
                "INSERT INTO workflows (id, owner_user_id, name, description, definition_json, is_template) "
                "VALUES (:id, NULL, :name, :description, :definition_json::jsonb, true) "
                "ON CONFLICT DO NOTHING"
            ), {"id": str(uuid.uuid4()), "name": name, "description": f"Pre-built {name} template",
                "definition_json": json.dumps(data)})

    def downgrade():
        op.get_bind().execute(sa.text("DELETE FROM workflows WHERE is_template = true"))
    ```

    IMPORTANT: owner_user_id=NULL is valid because the column was created nullable in migration 009.

```python
@tool
def fill_form(
    name: str | None = None,
    description: str | None = None,
    artifact_type: str | None = None,
    required_permissions: list[str] | None = None,
    model_alias: str | None = None,
    system_prompt: str | None = None,
    handler_module: str | None = None,
    sandbox_required: bool | None = None,
    entry_point: str | None = None,
    url: str | None = None,
    version: str | None = None,
) -> str:
    """
    Fill one or more form fields in the artifact creation form.
    Only provide the fields you want to set or change — omitted fields are unchanged.
    After calling this tool, the user will see the form fields update live.
    """
    # Build a summary of what was filled for the tool return message
    filled = {
        k: v for k, v in {
            "name": name, "description": description, "artifact_type": artifact_type,
            "required_permissions": required_permissions, "model_alias": model_alias,
            "system_prompt": system_prompt, "handler_module": handler_module,
            "sandbox_required": sandbox_required, "entry_point": entry_point,
            "url": url, "version": version,
        }.items() if v is not None
    }
    return f"Filled {len(filled)} field(s): {', '.join(filled.keys())}"
```

```python
from sqlalchemy import func

@router.get("/check-name")
async def check_agent_name(
    name: str = Query(..., min_length=1),
    user: UserContext = Depends(_require_registry_manager),
    session: AsyncSession = Depends(get_db),
) -> dict[str, bool]:
    """Returns {"available": true/false} for the given agent name (case-insensitive)."""
    async with session.begin():
        count = await session.scalar(
            select(func.count()).where(
                func.lower(AgentDefinition.name) == name.lower(),
                AgentDefinition.is_active == True,
            )
        )
    return {"available": count == 0}
```

**Step 1: Create frontend/src/app/api/admin/[type]/check-name/route.ts**

```typescript
import { auth } from "@/auth";
import { NextRequest, NextResponse } from "next/server";

const ALLOWED_TYPES = new Set(["agents", "tools", "skills", "mcp-servers"]);

export async function GET(
  req: NextRequest,
  { params }: { params: Promise<{ type: string }> }
) {
  const session = await auth();
  if (!session?.user) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const { type } = await params;
  if (!ALLOWED_TYPES.has(type)) {
    return NextResponse.json({ error: "Invalid type" }, { status: 400 });
  }

  const name = req.nextUrl.searchParams.get("name");
  if (!name) return NextResponse.json({ error: "name required" }, { status: 400 });

  const token = (session as unknown as Record<string, unknown>).access_token as string | undefined;
  const backendUrl = process.env.BACKEND_INTERNAL_URL ?? "http://backend:8000";
  const res = await fetch(
    `${backendUrl}/api/admin/${type}/check-name?name=${encodeURIComponent(name)}`,
    { headers: { Authorization: `Bearer ${token ?? ""}` }, cache: "no-store" }
  );
  const data: unknown = await res.json();
  return NextResponse.json(data, { status: res.status });
}
```

**Step 2: Create frontend/src/components/admin/artifact-wizard-name-check.tsx**

Controlled input that debounces 300ms, shows spinner while in-flight, then shows checkmark (available) or X (taken) per the CONTEXT.md decisions:

```typescript
"use client";
import { useEffect, useRef, useState } from "react";

interface Props {
  artifactType: string | null;  // "agent" | "tool" | "skill" | "mcp_server"
  value: string;
  onChange: (v: string) => void;
  onAvailabilityChange: (available: boolean | null) => void;  // null = in-flight
  disabled?: boolean;
  onBlur?: () => void;
}

const TYPE_TO_PATH: Record<string, string> = {
  agent: "agents", tool: "tools", skill: "skills", mcp_server: "mcp-servers",
};

export function ArtifactWizardNameCheck({ artifactType, value, onChange, onAvailabilityChange, disabled, onBlur }: Props) {
  const [status, setStatus] = useState<"idle" | "checking" | "available" | "taken">("idle");
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  useEffect(() => {
    if (!value || !artifactType) { setStatus("idle"); onAvailabilityChange(null); return; }
    setStatus("checking");
    onAvailabilityChange(null);
    if (timerRef.current) clearTimeout(timerRef.current);
    timerRef.current = setTimeout(async () => {
      const path = TYPE_TO_PATH[artifactType];
      if (!path) { setStatus("idle"); return; }
      try {
        const res = await fetch(`/api/admin/${path}/check-name?name=${encodeURIComponent(value)}`);
        const data = (await res.json()) as { available?: boolean };
        const avail = data.available ?? false;
        setStatus(avail ? "available" : "taken");
        onAvailabilityChange(avail);
      } catch { setStatus("idle"); onAvailabilityChange(null); }
    }, 300);
    return () => { if (timerRef.current) clearTimeout(timerRef.current); };
  }, [value, artifactType, onAvailabilityChange]);

  const badge =
    status === "checking" ? <span className="text-gray-400 text-xs ml-2">checking...</span> :
    status === "available" ? <span className="text-green-600 text-xs ml-2">✓ available</span> :
    status === "taken" ? <span className="text-red-600 text-xs ml-2">✗ taken</span> :
    null;

  return (
    <div className="flex items-center gap-2">
      <input
        type="text"
        value={value}
        onChange={(e) => onChange(e.target.value)}
        onBlur={onBlur}
        disabled={disabled}
        placeholder="artifact-name (lowercase, hyphens)"
        className="flex-1 px-3 py-2 border border-gray-300 rounded-md text-sm focus:outline-none focus:ring-2 focus:ring-blue-500 disabled:bg-gray-50"
      />
      {badge}
    </div>
  );
}
```

```typescript
export interface FormState {
  name: string;
  description: string;
  version: string;
  // Agent fields
  model_alias: string;
  system_prompt: string;
  // Tool + Skill fields
  required_permissions: string[];
  sandbox_required: boolean;
  handler_module: string;   // Tool
  entry_point: string;       // Skill
  // MCP Server fields
  url: string;
  auth_token: string;
}
```

Form sections (in order, single scrollable page):

**Section 1 — Artifact Type:** 4 radio buttons (Agent, Tool, Skill, MCP Server). Selecting one resets type-specific fields.

**Section 2 — Template / Clone:**
- "Start Blank" button: clears all fields
- "Clone Existing" button: opens CloneArtifactModal; on select, pre-fills all fields and appends `_copy` to name
- Template cards from ArtifactWizardTemplates component

**Section 3 — Common Fields:**
- Name: ArtifactWizardNameCheck component. Blur validation (per CONTEXT.md): error shown after leaving field, not while typing.
- Description: textarea
- Version: text input (default "1.0.0")

**Section 4 — Type-specific Fields:**

*Agent:* model_alias dropdown (`blitz/master`, `blitz/fast`, `blitz/coder`, `blitz/summarizer`) + system_prompt textarea

*Tool:* required_permissions multi-select checklist + sandbox_required checkbox + handler_module text input
- Permission strings list (hardcoded known permissions): `tool:email`, `tool:calendar`, `tool:project`, `tool:crm`, `mcp:read`, `mcp:write`, `registry:read`, `registry:manage`, `memory:read`, `memory:write`, `admin:read`, `admin:write`

*Skill:* required_permissions multi-select checklist + entry_point text input + sandbox_required checkbox

*MCP Server:* url input (type="url") + auth_token password input (optional)

**Section 5 — JSON Preview:**
```typescript
<pre className="bg-gray-50 border border-gray-200 rounded-md p-3 text-xs font-mono overflow-auto max-h-48">
  {JSON.stringify(buildSubmitPayload(formState, artifactType), null, 2)}
</pre>
```

**Section 6 — Actions:**
- Cancel button: calls `onCancel()` prop (resets form to blank)
- Submit button: disabled when `!nameAvailable || isSubmitting || !formState.name`; calls `onSubmit()`

AI-filled field pulse animation: When `aiFilledFields` prop contains a field name, add `animate-pulse bg-blue-50` class to that field's wrapper for 1500ms then remove. Use `useEffect` watching `aiFilledFields`.

Per CONTEXT.md decisions: "all fields pulse simultaneously — single animation frame, no stagger."

Props interface:
```typescript
interface ArtifactWizardFormProps {
  formState: FormState;
  artifactType: string | null;
  aiFilledFields: Set<string>;         // fields to pulse (from Task 2a parent)
  onFormChange: (updates: Partial<FormState>) => void;
  onArtifactTypeChange: (type: string) => void;
  onCancel: () => void;
  onSubmit: () => void;
  isSubmitting: boolean;
}
```

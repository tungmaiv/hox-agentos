Backend API endpoints (all under /api/storage):
```typescript
// Files
POST   /api/storage/files/upload          // FormData: file + optional folder_id
// Response: StorageFileResponse | DedupResponse
GET    /api/storage/files?folder_id=&page=&limit=  // → { files: StorageFile[], total: number }
GET    /api/storage/files/{file_id}       // → StorageFile with download_url
DELETE /api/storage/files/{file_id}
POST   /api/storage/files/{file_id}/add-to-memory  // → { status: "queued" }

// Folders
POST   /api/storage/folders               // body: { name, parent_folder_id? }
GET    /api/storage/folders               // → StorageFolder[]
DELETE /api/storage/folders/{folder_id}

// Shares
POST   /api/storage/shares                // body: { resource_type, resource_id, shared_with_user_id, permission }
GET    /api/storage/shares/{file_id}      // → StorageShare[]
PATCH  /api/storage/shares/{share_id}     // body: { permission }
DELETE /api/storage/shares/{share_id}
GET    /api/storage/shared-with-me        // → { files: StorageFile[], folders: StorageFolder[] }

// User search
GET    /api/storage/users/search?q=email  // → [{ id, email, display_name }]
```

TypeScript types to define in types.ts:
```typescript
interface StorageFile {
  id: string;
  name: string;
  mime_type: string;
  size_bytes: number;
  content_hash: string;
  folder_id: string | null;
  owner_user_id: string;
  in_memory: boolean;
  created_at: string;
  updated_at: string;
  download_url?: string;
}

interface StorageFolder {
  id: string;
  name: string;
  parent_folder_id: string | null;
  owner_user_id: string;
  created_at: string;
}

interface StorageShare {
  id: string;
  file_id: string | null;
  folder_id: string | null;
  shared_with_user_id: string;
  shared_by_user_id: string;
  permission: "READ" | "WRITE" | "ADMIN";
  created_at: string;
}

interface UploadProgress {
  file: File;
  progress: number;   // 0-100
  status: "uploading" | "done" | "duplicate" | "error";
  dedupInfo?: { existing_file_id: string; existing_file_name: string };
}
```

         ```python
         async def _invoke_agent(self, msg: InternalMessage) -> InternalMessage:
             from agents.master_agent import run_conversation
             from core.context import user_id_ctx, conversation_id_ctx

             # Set contextvars for memory isolation (same pattern as gateway/runtime.py)
             user_id_ctx.set(str(msg.user_id))
             conversation_id_ctx.set(str(msg.conversation_id))

             try:
                 response_text = await asyncio.wait_for(
                     run_conversation(
                         user_id=msg.user_id,
                         conversation_id=msg.conversation_id,
                         messages=[{"role": "user", "content": msg.text or ""}],
                     ),
                     timeout=60.0,  # 60 second timeout per locked decision
                 )
             except asyncio.TimeoutError:
                 logger.error("channel_agent_timeout", channel=msg.channel, user_id=str(msg.user_id))
                 response_text = "Sorry, I couldn't process your request. Please try again."
             except Exception as exc:
                 logger.error("channel_agent_error", channel=msg.channel, error=str(exc))
                 response_text = "Sorry, I couldn't process your request. Please try again."

             return self._make_reply(msg, response_text)
         ```

         ```python
         async def _handle_channel_output_node(config, state):
             from channels.gateway import ChannelGateway
             from channels.models import InternalMessage
             from api.routes.channels import get_channel_gateway

             channel = config.get("channel", "web")
             template = config.get("template", "{output}")
             output = state.get("current_output")
             user_context = state.get("user_context", {})

             try:
                 message = template.format(output=output)
             except (KeyError, ValueError):
                 message = str(output)

             if channel == "web":
                 return {"channel": channel, "message": message, "sent": True}

             gateway = get_channel_gateway()
             msg = InternalMessage(
                 direction="outbound",
                 channel=channel,
                 external_user_id=user_context.get("external_user_id", ""),
                 text=message,
             )
             await gateway.send_outbound(msg)
             return {"channel": channel, "message": message, "sent": True}
         ```

**Step 1: Add Celery reindex task to `scheduler/tasks/embedding.py`**

```python
@celery_app.task(name="blitz.reindex_memory")
def reindex_memory_task() -> None:
    """
    Re-embed all memory facts and episodes from source text.

    WARNING: Deletes all embedding vectors before re-embedding.
    Run only when called explicitly via POST /api/admin/memory/reindex?confirm=true.
    """
    import asyncio

    async def _run() -> None:
        from core.db import async_session
        from core.models.memory_long_term import MemoryEpisode
        from core.models.memory_short_term import MemoryFact
        from memory.embeddings import SidecarEmbeddingProvider
        from sqlalchemy import select, update

        provider = SidecarEmbeddingProvider()
        logger.info("memory_reindex_started")

        async with async_session() as session:
            async with session.begin():
                # Fetch all facts with source text
                result = await session.execute(
                    select(MemoryFact.id, MemoryFact.content)
                )
                facts = result.all()

            # Re-embed in batches of 32
            batch_size = 32
            for i in range(0, len(facts), batch_size):
                batch = facts[i : i + batch_size]
                texts = [row.content for row in batch]
                vectors = await provider.embed(texts)
                async with session.begin():
                    for (fact_id, _), vector in zip(batch, vectors, strict=True):
                        await session.execute(
                            update(MemoryFact)
                            .where(MemoryFact.id == fact_id)
                            .values(embedding=vector)
                        )

            # Do the same for episodes
            result = await session.execute(
                select(MemoryEpisode.id, MemoryEpisode.summary)
            )
            episodes = result.all()
            for i in range(0, len(episodes), batch_size):
                batch = episodes[i : i + batch_size]
                texts = [row.summary for row in batch]
                vectors = await provider.embed(texts)
                async with session.begin():
                    for (ep_id, _), vector in zip(batch, vectors, strict=True):
                        await session.execute(
                            update(MemoryEpisode)
                            .where(MemoryEpisode.id == ep_id)
                            .values(embedding=vector)
                        )

        logger.info("memory_reindex_complete", facts=len(facts), episodes=len(episodes))

    asyncio.run(_run())
```

**Step 2: Create `backend/api/routes/admin_memory.py`**

```python
"""Admin memory management routes."""
from pydantic import BaseModel, model_validator
from fastapi import APIRouter, Depends, HTTPException

import structlog
from security.deps import get_current_user
from security.rbac import require_roles
from core.models.user import UserContext
from scheduler.tasks.embedding import reindex_memory_task

logger = structlog.get_logger(__name__)
router = APIRouter(prefix="/api/admin/memory", tags=["admin-memory"])


class ReindexRequest(BaseModel):
    confirm: bool

    @model_validator(mode="after")
    def must_confirm(self) -> "ReindexRequest":
        if not self.confirm:
            raise ValueError("confirm must be true to proceed with reindex")
        return self


class ReindexResponse(BaseModel):
    job_id: str
    message: str


@router.post("/reindex", response_model=ReindexResponse, status_code=202)
async def trigger_reindex(
    body: ReindexRequest,
    user: UserContext = Depends(get_current_user),
) -> ReindexResponse:
    """
    Enqueue a full memory reindex job.

    Deletes all embedding vectors and re-embeds from source text.
    Requires admin role. Requires confirm=true in body.
    """
    require_roles(user, ["admin"])
    task = reindex_memory_task.delay()
    logger.info("memory_reindex_enqueued", job_id=task.id, user_id=str(user.user_id))
    return ReindexResponse(
        job_id=task.id,
        message="Reindex job enqueued. All memory vectors will be re-embedded.",
    )
```

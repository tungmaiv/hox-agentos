```python
def test_get_llm_returns_chatopenai_pointing_at_litellm():
    """get_llm() must return a ChatOpenAI client pointed at LiteLLM proxy, not a provider."""
    from core.config import get_llm
    llm = get_llm("blitz/master")
    # Find the base_url attribute (may be openai_api_base or base_url depending on langchain version)
    base_url = getattr(llm, "openai_api_base", None) or getattr(llm, "base_url", None) or ""
    base_url_str = str(base_url)
    assert "4000" in base_url_str or "litellm" in base_url_str, (
        f"Expected LiteLLM URL (port 4000 or 'litellm'), got: {base_url_str}"
    )


def test_get_llm_uses_correct_model_alias():
    """Each alias maps to the correct sanitized model name sent to LiteLLM."""
    from core.config import get_llm
    cases = {
        "blitz/master": "blitz-master",
        "blitz/fast": "blitz-fast",
        "blitz/coder": "blitz-coder",
        "blitz/summarizer": "blitz-summarizer",
    }
    for alias, expected_model in cases.items():
        llm = get_llm(alias)
        actual_model = getattr(llm, "model_name", None) or getattr(llm, "model", None) or ""
        assert actual_model == expected_model, (
            f"alias='{alias}': expected model='{expected_model}', got='{actual_model}'"
        )


def test_get_llm_unknown_alias_passes_through():
    """An unknown alias is passed through unchanged — LiteLLM handles the error, not backend code."""
    from core.config import get_llm
    llm = get_llm("custom/my-model")
    actual_model = getattr(llm, "model_name", None) or getattr(llm, "model", None) or ""
    assert actual_model == "custom/my-model", (
        f"Expected model to pass through unchanged, got: '{actual_model}'"
    )
```

```typescript
"use client";
import { signOut, useSession } from "next-auth/react";

export function SignOutButton() {
  const { data: session } = useSession();

  async function handleSignOut(): Promise<void> {
    // For Keycloak users: call Keycloak end-session endpoint to fully revoke SSO session
    // This is critical for shared office machines — partial logout is a security risk
    if (session?.authProvider === "keycloak" && session?.idToken) {
      const keycloakIssuer = process.env.NEXT_PUBLIC_KEYCLOAK_ISSUER ?? "";
      if (keycloakIssuer) {
        const logoutUrl = new URL(
          `${keycloakIssuer}/protocol/openid-connect/logout`
        );
        logoutUrl.searchParams.set("id_token_hint", session.idToken);
        logoutUrl.searchParams.set(
          "post_logout_redirect_uri",
          `${window.location.origin}/login?signedOut=true`
        );

        // Clear next-auth session first, then redirect to Keycloak end-session
        await signOut({ redirect: false });
        window.location.href = logoutUrl.toString();
        return;
      }
    }

    // For local auth users (or if Keycloak issuer not configured):
    // Just clear next-auth cookies and redirect to /login
    await signOut({ callbackUrl: "/login?signedOut=true" });
  }

  return (
    <button
      onClick={() => void handleSignOut()}
      className="text-sm text-gray-500 hover:text-gray-900 transition-colors"
    >
      Sign out
    </button>
  );
}
```

**IMPORTANT:** `NEXT_PUBLIC_KEYCLOAK_ISSUER` must be set in `frontend/.env.local` for Keycloak logout to work. It should be the same issuer URL used in auth.ts (`KEYCLOAK_ISSUER`). Check if this env var already exists; if not, add `NEXT_PUBLIC_KEYCLOAK_ISSUER` alongside the existing `KEYCLOAK_ISSUER` in the frontend env config.

Actually, looking at this more carefully: `session.idToken` and `session.authProvider` are server-side session fields. next-auth does NOT expose arbitrary session fields to the client by default. The `session()` callback controls what the client sees. Since we're adding `idToken` and `authProvider` to the session callback return, they WILL be available on the client via `useSession()`. This is safe because `idToken` is the OIDC identity token (used only for logout), not the access token.

```typescript
"use client";
/**
 * Auth error toast notifications + session error detection.
 *
 * Detects session errors from next-auth (SessionExpired, RefreshAccessTokenError)
 * and auto-redirects to /login with a toast notification.
 * Also renders the Sonner Toaster for all auth-related toasts.
 */
import { useEffect } from "react";
import { useSession, signOut } from "next-auth/react";
import { Toaster, toast } from "sonner";
import { usePathname } from "next/navigation";

export function AuthErrorToasts() {
  const { data: session } = useSession();
  const pathname = usePathname();

  useEffect(() => {
    // Don't trigger on login page (avoid redirect loop)
    if (pathname === "/login") return;

    if (
      session?.error === "SessionExpired" ||
      session?.error === "RefreshAccessTokenError"
    ) {
      // Show toast notification
      toast.error("Your session has expired. Please sign in again.", {
        duration: 4000,
      });

      // Auto-redirect to /login after a brief delay for toast visibility
      const timer = setTimeout(() => {
        void signOut({
          callbackUrl: `/login?callbackUrl=${encodeURIComponent(pathname)}&error=${session.error}`,
        });
      }, 1500);

      return () => clearTimeout(timer);
    }
  }, [session?.error, pathname]);

  return <Toaster position="top-right" richColors />;
}
```

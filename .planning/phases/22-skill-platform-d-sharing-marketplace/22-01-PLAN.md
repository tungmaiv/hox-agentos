@router.patch("/{skill_id}/promote")
async def toggle_skill_promoted(
    skill_id: UUID,
    user: UserContext = Depends(_require_registry_manager),
    session: AsyncSession = Depends(get_db),
) -> SkillDefinitionResponse:
    """Toggle the is_promoted flag on a skill.

    Calling this endpoint twice returns the skill to its previous promoted state.
    Requires registry:manage permission.
    """
    from sqlalchemy import select

    result = await session.execute(
        select(SkillDefinition).where(SkillDefinition.id == skill_id)
    )
    skill = result.scalar_one_or_none()
    if skill is None:
        raise HTTPException(status_code=404, detail="Skill not found")

    skill.is_promoted = not skill.is_promoted
    await session.commit()
    await session.refresh(skill)

    logger.info(
        "admin_skill_promoted_toggled",
        skill_id=str(skill_id),
        skill_name=skill.name,
        is_promoted=skill.is_promoted,
        user_id=str(user["user_id"]),
    )
    return SkillDefinitionResponse.model_validate(skill)
```

@router.get("/shared-with-me")
async def list_skills_shared_with_me(
    user: UserContext = Depends(_require_chat),
    session: AsyncSession = Depends(get_user_db),
) -> list[SkillListItem]:
    """List skills explicitly shared with the current user via user_artifact_permissions.

    Additive catalog section — shows skills directly shared with this user_id.
    Does not restrict visibility of the main catalog.
    """
    from core.models.user_artifact_permission import UserArtifactPermission

    # Find all skill IDs shared with this user
    perm_result = await session.execute(
        select(UserArtifactPermission).where(
            UserArtifactPermission.artifact_type == "skill",
            UserArtifactPermission.user_id == user["user_id"],
            UserArtifactPermission.allowed == True,  # noqa: E712
            UserArtifactPermission.status == "active",
        )
    )
    perms = perm_result.scalars().all()
    if not perms:
        return []

    skill_ids = [p.artifact_id for p in perms]
    skill_result = await session.execute(
        select(SkillDefinition).where(
            SkillDefinition.id.in_(skill_ids),
            SkillDefinition.status == "active",
            SkillDefinition.is_active == True,  # noqa: E712
        )
    )
    skills = skill_result.scalars().all()
    logger.info(
        "user_shared_skills_listed",
        user_id=str(user["user_id"]),
        count=len(skills),
    )
    return [SkillListItem.model_validate(s) for s in skills]


@router.get("/{skill_id}/export")
async def export_user_skill(
    skill_id: UUID,
    user: UserContext = Depends(_require_chat),
    session: AsyncSession = Depends(get_user_db),
) -> StreamingResponse:
    """Export a skill as an agentskills.io-compliant ZIP.

    Only active skills can be exported. Returns a zip archive with SKILL.md,
    MANIFEST.json, and optional scripts/ and references/ directories.
    """
    result = await session.execute(
        select(SkillDefinition).where(
            SkillDefinition.id == skill_id,
            SkillDefinition.status == "active",
            SkillDefinition.is_active == True,  # noqa: E712
        )
    )
    skill = result.scalar_one_or_none()
    if skill is None:
        raise HTTPException(status_code=404, detail="Skill not found or not active")

    zip_bytes = build_skill_zip(skill)
    filename = f"{skill.name}-{skill.version}.zip"

    logger.info(
        "user_skill_exported",
        skill_id=str(skill_id),
        skill_name=skill.name,
        filename=filename,
        user_id=str(user["user_id"]),
    )
    return StreamingResponse(
        zip_bytes,
        media_type="application/zip",
        headers={"Content-Disposition": f'attachment; filename="{filename}"'},
    )
```

```python
"""
Admin endpoints for skill sharing — grant and revoke per-user access.

POST   /api/admin/skills/{skill_id}/share              — share with a specific user
DELETE /api/admin/skills/{skill_id}/share/{user_id}   — revoke sharing
GET    /api/admin/skills/{skill_id}/shares             — list all users with access

Security: requires registry:manage permission (Gate 2 RBAC).

These endpoints create/delete rows in user_artifact_permissions with
artifact_type='skill'. Sharing is additive — it creates a "Shared with me"
catalog section for the recipient without restricting global skill visibility.
"""
from uuid import UUID

import structlog
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy import delete, select
from sqlalchemy.exc import IntegrityError
from sqlalchemy.ext.asyncio import AsyncSession

from core.db import get_db
from core.models.user import UserContext
from core.models.user_artifact_permission import UserArtifactPermission
from core.schemas.registry import SkillShareEntry, SkillShareRequest
from security.deps import require_registry_manager

logger = structlog.get_logger(__name__)

router = APIRouter(prefix="/api/admin/skills", tags=["admin-skills-sharing"])


@router.post("/{skill_id}/share", status_code=201)
async def share_skill_with_user(
    skill_id: UUID,
    body: SkillShareRequest,
    user: UserContext = Depends(require_registry_manager),
    session: AsyncSession = Depends(get_db),
) -> SkillShareEntry:
    """Share a skill with a specific user.

    Creates a UserArtifactPermission row granting the user access.
    Returns 409 if the skill is already shared with this user.
    """
    perm = UserArtifactPermission(
        artifact_type="skill",
        artifact_id=skill_id,
        user_id=body.user_id,
        allowed=True,
        status="active",
    )
    session.add(perm)
    try:
        await session.commit()
        await session.refresh(perm)
    except IntegrityError:
        await session.rollback()
        raise HTTPException(
            status_code=409,
            detail="Skill is already shared with this user",
        )

    logger.info(
        "admin_skill_shared",
        skill_id=str(skill_id),
        target_user_id=str(body.user_id),
        admin_user_id=str(user["user_id"]),
    )
    return SkillShareEntry(user_id=perm.user_id, granted_at=perm.created_at)


@router.delete("/{skill_id}/share/{target_user_id}", status_code=204)
async def revoke_skill_share(
    skill_id: UUID,
    target_user_id: UUID,
    user: UserContext = Depends(require_registry_manager),
    session: AsyncSession = Depends(get_db),
) -> None:
    """Revoke a user's access to a shared skill.

    Deletes the UserArtifactPermission row. Returns 404 if no share exists.
    """
    result = await session.execute(
        select(UserArtifactPermission).where(
            UserArtifactPermission.artifact_type == "skill",
            UserArtifactPermission.artifact_id == skill_id,
            UserArtifactPermission.user_id == target_user_id,
        )
    )
    perm = result.scalar_one_or_none()
    if perm is None:
        raise HTTPException(status_code=404, detail="Share not found")

    await session.execute(
        delete(UserArtifactPermission).where(UserArtifactPermission.id == perm.id)
    )
    await session.commit()

    logger.info(
        "admin_skill_share_revoked",
        skill_id=str(skill_id),
        target_user_id=str(target_user_id),
        admin_user_id=str(user["user_id"]),
    )


@router.get("/{skill_id}/shares")
async def list_skill_shares(
    skill_id: UUID,
    user: UserContext = Depends(require_registry_manager),
    session: AsyncSession = Depends(get_db),
) -> list[SkillShareEntry]:
    """List all users the skill is shared with."""
    result = await session.execute(
        select(UserArtifactPermission).where(
            UserArtifactPermission.artifact_type == "skill",
            UserArtifactPermission.artifact_id == skill_id,
            UserArtifactPermission.allowed == True,  # noqa: E712
            UserArtifactPermission.status == "active",
        )
    )
    perms = result.scalars().all()
    return [SkillShareEntry(user_id=p.user_id, granted_at=p.created_at) for p in perms]
```

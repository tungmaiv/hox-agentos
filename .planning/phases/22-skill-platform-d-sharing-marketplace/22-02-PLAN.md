**Step 2: Add share-loading function**

Add `loadShares` function alongside `handleExport` and `handlePromote`:

```typescript
const loadShares = async (skillId: string) => {
  setSharesLoading(true);
  setShareError(null);
  try {
    const res = await fetch(`/api/admin/skills/${skillId}/shares`);
    if (!res.ok) {
      setShareError("Failed to load shares");
      return;
    }
    const data = (await res.json()) as SkillShareEntry[];
    setShares(data);
  } catch {
    setShareError("Network error loading shares");
  } finally {
    setSharesLoading(false);
  }
};
```

**Step 4: Add the sharing panel UI to the detail drawer**

In the detail panel section (where `SkillMetadataPanel` and the promoted toggle live), add the sharing panel below them:

```tsx
{/* Sharing panel */}
<div className="mt-3 pt-3 border-t border-gray-100">
  <p className="text-xs font-semibold text-gray-700 mb-2">Shared with</p>

  {sharesLoading && (
    <p className="text-xs text-gray-400">Loading...</p>
  )}

  {!sharesLoading && shares.length === 0 && (
    <p className="text-xs text-gray-400">Not shared with any specific users.</p>
  )}

  {shares.map((share) => (
    <div key={share.user_id} className="flex items-center justify-between mb-1">
      <span className="text-xs font-mono text-gray-600 truncate max-w-xs">
        {share.user_id}
      </span>
      <button
        onClick={async () => {
          const res = await fetch(
            `/api/admin/skills/${(item as SkillDefinition).id}/share/${share.user_id}`,
            { method: "DELETE" }
          );
          if (res.ok || res.status === 204) {
            setShares((prev) => prev.filter((s) => s.user_id !== share.user_id));
          }
        }}
        className="ml-2 text-xs text-red-500 hover:text-red-700 shrink-0"
      >
        Revoke
      </button>
    </div>
  ))}

  {/* Add share input */}
  <div className="flex gap-1 mt-2">
    <input
      type="text"
      value={shareInput}
      onChange={(e) => setShareInput(e.target.value)}
      placeholder="User UUID..."
      className="flex-1 text-xs border border-gray-300 rounded px-2 py-1 focus:outline-none focus:ring-1 focus:ring-blue-500"
    />
    <button
      onClick={async () => {
        if (!shareInput.trim()) return;
        setShareError(null);
        const res = await fetch(
          `/api/admin/skills/${(item as SkillDefinition).id}/share`,
          {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ user_id: shareInput.trim() }),
          }
        );
        if (res.status === 201) {
          const newShare = (await res.json()) as SkillShareEntry;
          setShares((prev) => [...prev, newShare]);
          setShareInput("");
        } else if (res.status === 409) {
          setShareError("Already shared with this user");
        } else {
          setShareError("Failed to share skill");
        }
      }}
      className="px-2 py-1 text-xs bg-blue-600 text-white rounded hover:bg-blue-700"
    >
      Share
    </button>
  </div>

  {shareError && (
    <p className="text-xs text-red-500 mt-1">{shareError}</p>
  )}
</div>
```

Note: `item` here refers to the currently selected item in the detail panel. Check the actual variable name in the existing detail panel render code and use the correct one.

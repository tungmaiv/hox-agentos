3. **Add useEffect to fetch count** — After the existing hooks, add:
```tsx
React.useEffect(() => {
  const fetchPending = async () => {
    try {
      const token = (session as Record<string, unknown>)?.access_token as string | undefined;
      if (!token) return;
      const res = await fetch(
        `${process.env.NEXT_PUBLIC_API_URL ?? ""}/api/registry?type=skill&status=pending_activation`,
        {
          headers: { Authorization: `Bearer ${token}` },
          cache: "no-store",
        }
      );
      if (!res.ok) return;
      const data: Array<{id: string; name: string}> = await res.json();
      setPendingCount(data.length);
      setPendingSkills(data);
    } catch {
      // Bell is non-critical — swallow errors
    }
  };
  fetchPending();
}, [session]);
```

**Fetch function — replace `fetchSkills`:**
```typescript
const fetchSkills = useCallback(async (q: string, append = false, currentCursor = 0) => {
  if (!append) setLoading(true);
  else setLoadingMore(true);
  setError(null);
  try {
    const params = new URLSearchParams({ limit: "20", cursor: String(currentCursor) });
    if (q.trim()) params.set("q", encodeURIComponent(q.trim()));
    const response = await fetch(`/api/skill-repos/browse?${params.toString()}`);
    if (!response.ok) throw new Error(`Error ${response.status}`);
    const data: unknown = await response.json();
    if (!Array.isArray(data)) throw new Error("Unexpected response");
    const items = data as SkillBrowseItem[];
    if (append) {
      setSkills((prev) => [...prev, ...items]);
    } else {
      setSkills(items);
      setCursor(0);
    }
    setHasMore(items.length === 20);
  } catch (err) {
    setError(err instanceof Error ? err.message : "Failed to load skills");
  } finally {
    setLoading(false);
    setLoadingMore(false);
  }
}, []);
```

**Load More handler:**
```typescript
const handleLoadMore = useCallback(() => {
```

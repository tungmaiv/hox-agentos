## LiteLLM model management API

Confirmed endpoints (from research):
- `POST {litellm_url}/model/new` — add model
- `POST {litellm_url}/model/update` — update model
- `DELETE {litellm_url}/model/delete` — delete model
- `GET {litellm_url}/model/info` — list models

Authorization: `Authorization: Bearer {settings.litellm_master_key}` header required.

**LIMITATION (must show in UI):** LiteLLM stores models in DB only if DB configured. Without DB, changes are in-memory and lost on restart. Add visible disclaimer in the LLM admin page.

## LLM config UI behavior

The LLM admin page (`/admin/system/llm`) should:
1. On load: fetch current model list from `/api/admin/llm/models` (backend fetches from LiteLLM `/model/info`)
2. Show table: model alias | provider model | status
3. "Add Model" button: opens inline form with fields: alias (e.g. blitz/master), provider_model (e.g. claude-sonnet-4-6), api_base (optional)
4. "Save" calls POST /api/admin/llm/models → calls LiteLLM /model/new
5. Show disclaimer banner above table: "Model changes apply immediately. For persistence across restarts, also update infra/litellm/config.yaml."

    ```python
    """Admin LLM model configuration — manages LiteLLM proxy models via API."""
    from fastapi import APIRouter, Depends, HTTPException
    from pydantic import BaseModel
    import httpx
    import structlog
    from core.config import settings
    from security.deps import get_current_user
    from security.rbac import check_permission

    logger = structlog.get_logger(__name__)
    router = APIRouter(prefix="/admin/llm", tags=["admin-llm"])

    class AddModelRequest(BaseModel):
        model_alias: str
        provider_model: str
        api_base: str | None = None
        api_key: str | None = None

    class ModelInfo(BaseModel):
        model_alias: str
        provider_model: str | None = None
        api_base: str | None = None

    class LLMConfigResponse(BaseModel):
        models: list[ModelInfo]
        litellm_available: bool

    def _litellm_headers() -> dict[str, str]:
        return {"Authorization": f"Bearer {settings.litellm_master_key}"}

    @router.get("/models", response_model=LLMConfigResponse)
    async def get_models(current_user=Depends(get_current_user)):
        check_permission(current_user, "tool:admin")
        try:
            async with httpx.AsyncClient(timeout=5.0) as client:
                resp = await client.get(f"{settings.litellm_url}/model/info", headers=_litellm_headers())
                resp.raise_for_status()
                data = resp.json()
                models = []
                for entry in data.get("data", []):
                    models.append(ModelInfo(
                        model_alias=entry.get("model_name", ""),
                        provider_model=entry.get("litellm_params", {}).get("model"),
                        api_base=entry.get("litellm_params", {}).get("api_base"),
                    ))
                return LLMConfigResponse(models=models, litellm_available=True)
        except (httpx.ConnectError, httpx.TimeoutException) as exc:
            logger.warning("litellm_unavailable", error=str(exc))
            return LLMConfigResponse(models=[], litellm_available=False)

    @router.post("/models", status_code=201)
    async def add_model(request: AddModelRequest, current_user=Depends(get_current_user)):
        check_permission(current_user, "tool:admin")
        payload = {
            "model_name": request.model_alias,
            "litellm_params": {
                "model": request.provider_model,
                **({"api_base": request.api_base} if request.api_base else {}),
                **({"api_key": request.api_key} if request.api_key else {}),
            }
        }
        async with httpx.AsyncClient(timeout=10.0) as client:
            resp = await client.post(f"{settings.litellm_url}/model/new", json=payload, headers=_litellm_headers())
            if resp.status_code not in (200, 201):
                raise HTTPException(status_code=502, detail=f"LiteLLM error: {resp.text}")
        logger.info("llm_model_added", alias=request.model_alias, model=request.provider_model)
        return {"status": "added", "model_alias": request.model_alias}

    @router.delete("/models/{model_alias}", status_code=204)
    async def delete_model(model_alias: str, current_user=Depends(get_current_user)):
        check_permission(current_user, "tool:admin")
        async with httpx.AsyncClient(timeout=10.0) as client:
            resp = await client.delete(
                f"{settings.litellm_url}/model/delete",
                json={"id": model_alias},
                headers=_litellm_headers(),
            )
            if resp.status_code not in (200, 204):
                raise HTTPException(status_code=502, detail=f"LiteLLM error: {resp.text}")
        logger.info("llm_model_deleted", alias=model_alias)
    ```

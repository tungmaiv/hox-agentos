## CRITICAL PITFALL: Subprocess hanging

StdioMCPClient subprocess can hang if server exits prematurely. Always wrap call_tool in `asyncio.wait_for(..., timeout=30.0)`. Log process exit code on error.

    ```python
    """Stdio-transport MCP client for CLI-installed servers (Context7, Fetch, Filesystem)."""
    import asyncio
    from typing import Any
    import structlog
    from mcp import ClientSession, StdioServerParameters
    from mcp.client.stdio import stdio_client

    logger = structlog.get_logger(__name__)

    class StdioMCPClient:
        """Subprocess-based MCP client using stdio transport."""

        def __init__(self, command: str, args: list[str], env: dict[str, str] | None = None) -> None:
            self._params = StdioServerParameters(command=command, args=args, env=env)
            self._command = command

        async def list_tools(self) -> list[dict[str, Any]]:
            async with stdio_client(self._params) as (read, write):
                async with ClientSession(read, write) as session:
                    await session.initialize()
                    response = await session.list_tools()
                    return [t.model_dump() for t in response.tools]

        async def call_tool(self, tool_name: str, arguments: dict[str, Any], timeout: float = 30.0) -> dict[str, Any]:
            async def _call() -> dict[str, Any]:
                async with stdio_client(self._params) as (read, write):
                    async with ClientSession(read, write) as session:
                        await session.initialize()
                        result = await session.call_tool(tool_name, arguments)
                        return result.model_dump()

            try:
                return await asyncio.wait_for(_call(), timeout=timeout)
            except asyncio.TimeoutError:
                logger.warning("stdio_mcp_timeout", command=self._command, tool=tool_name, timeout=timeout)
                raise
    ```

    **Create backend/mcp/installer.py:**
    ```python
    """MCPInstaller: install npm/pip packages for stdio MCP servers."""
    import asyncio
    import structlog
    from typing import Literal

    logger = structlog.get_logger(__name__)

    class MCPInstallError(Exception):
        pass

    class MCPInstaller:
        async def install(self, package_manager: Literal["npm", "pip"], package_name: str) -> None:
            if package_manager == "npm":
                cmd = ["npm", "install", "-g", package_name]
            elif package_manager == "pip":
                cmd = ["pip", "install", package_name]
            else:
                raise ValueError(f"Unknown package manager: {package_manager}. Use 'npm' or 'pip'.")

            logger.info("mcp_install", package_manager=package_manager, package=package_name)
            proc = await asyncio.create_subprocess_exec(
                *cmd,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
            )
            stdout, stderr = await proc.communicate()
            if proc.returncode != 0:
                raise MCPInstallError(
                    f"{package_manager} install {package_name} failed (exit {proc.returncode}): {stderr.decode()}"
                )
            logger.info("mcp_install_success", package_manager=package_manager, package=package_name)

        async def is_installed(self, package_manager: Literal["npm", "pip"], package_name: str) -> bool:
            if package_manager == "npm":
                cmd = ["npm", "list", "-g", package_name]
            elif package_manager == "pip":
                cmd = ["pip", "show", package_name]
            else:
                return False
            proc = await asyncio.create_subprocess_exec(
                *cmd,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
            )
            await proc.communicate()
            return proc.returncode == 0
    ```
